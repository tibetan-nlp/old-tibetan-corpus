གུང་རྟོན་བཀུམ། བློན་ཆེ་ཁྲི་འབྲིང་འ་ཞ་ཡུལ་དུ་མཆིས་ཤིང། སྟག་ལ་རྒྱ་དུར་དུ་རྒྱའི་དམག་པོན་ཝང་ཞང་ཤོ་དང་གཡུལ་སྤྲད་དེ་རྒྱ་མང་པོ་བཀུཾ་བར་ལོ་ཆིག ། ༆། སྤྲེའུ་ལོ་ལ། བཙན་པོ་ཟྲིད་མདའ་ན་བཞུགས་ཤིང། བློན་ཅེ་ཁྲི་འབྲིང་གྱིས། འ་ཞ་ཡུལ་གྱི་སིལ་གུ་ཅིན་གྱི་འོ་ཀོལ་དུ་འ་ཞའི་མཁོས་བགྱིས། དགུན་འོ་བར་ཚལ་དུ་མང་ཉེན་བཞི་བརྩན་གྱིས་བསྡུས་ཏེ། བཙན་མོ་མང་མོ་རྗེ་ལས་མི་མང་པོ་བཀུག་ལོ་ཅིག ། ༆། བྱ་གགི་ལོ་ལ་བབ་སྟེ། བཙན་པོ་བལ་པོ་ན་བཞུགས་ཤིང། ཅེ་དོག་པན་གྱི་ཕོ་ཉ་ཕྱག་འཚལད། དགུན་པོ་བྲང་བྲག་མར་གྱི་ཙལ་ཀ་ན་བཞུགས་ཤིང། འདུན་མ་འོན་གྱི་འ་ག་ཙལ་ན་བཞུགས་པར་ལོ་གཅིག ། ༆ ། ཁྱི་ལོ་ལ་བབ་སྟེ། བཙན་པོ་དབྱརད་བྱང་རོལ་དུ་གཤེགས། དགུན་བློན་ཆེན་པོ་ཁྲི་འབྲིང་གྱིས། ཙོང་ཀ་ཆེ་ཆུང་དུ་དྲངས་ཏེ། རྒྱའི་དམག་པོན་ཆེན་པོ་ཐུག་པུ་ཤི་བཟུང། དེའི་དགུན་མགར་ལ་བཀྱོན་ཕབ་སྟེ་བཙན་པོ་ཕར་དུ་གཤེགས་པར་ལོ་ཆིག ། ༆། པགི་ལོ་ལ་བབ་སྟེ། བཙན་པོ་དབྱརད་ཕར་ནས་བལ་ཕོའི་བྲིའུ་ཏང་དུ་གཤེགས་ཤིང། རྒྱའི་ཕོ་ཉ་ཇེའུ་ཞང་ཤོ་ཕྱག་འཚལད། དགུན་བཙན་པོ་དོལད་གྱི་མར་མ་ན་བཞུགས་ཤིང། གློ་བ་ཉེ་བ་ཡིག་གཙང་དང་བྱ་སྒ་སྩལད། ཏོན་ཡ་བགོ་ཁ་གན་ཕྱག་འཚལད། སྒྲེགས་གྱི་བྱ་ཙལ་དུ་བཀྱོན་བབ་གྱི་ནོར་བརྩིས་པར་ལོ་གཅིག ། ༆། བྱི་བའི་ལོ་ལ་བབ་སྟེ། བཙན་པོ་དབྱརད་མོང་ཀར་ནས་ཆབ་སྲིད་ལ་ཤ་གུ་ཉིང་སུམ་ཁོལ་དུ་གཤེགས་ཤིང། ཏོན་ཡ་བགོ་ཁ་གན་དྲུ་གུ་ཡུལ་དུ་བཏང། སྟོན་བཙན་པོ་གཤེགས་ཏེ་ག་ཆུར་དྲངས། དགུན་བཙན་པོའི་པོ་བྲང་རྨ་བྱ་ཙལ་ན་བཞུགས། ཡུམ་ཁྲི་མ་ལོད་འོན་ཅང་དོ་ན་བཞུགས་ཤིང། འབའ་ད་ཤི་ཕྱག་འཚལད་པར་ལོ་ཆིག ། ༆། གླང་གྱི་ལོ་ལ་བབ་སྟེ། བཙན་པོ་དབྱརད་གསེར་ཞ་ན་བཞུགས་ཤིང། ཟོང་ཅུ་དང་ཐེའུ་ཆུར་དྲངས་ཏེ། ཡུམ་ཁྲི་མ་ལོད་སྟོད་གྱི་མོང་ན་བཞུགས་ཤིང། དགུན་བཙན་པོ་ཁྲི་རྩེ་ན་བཞུགས་ཅིང། དཔྱིད་ལྟབ་མར། ཟོང་ཆུ་དང། ཐེའུ་ཅུར་དྲངས་ཏེ། ཡུམ་ཁྲི་མ་ལོད། ངན་ལམ་ཙལ་སར་པ་ན་བཞུགས་ཤིང། འདུན་མ་གླག་གུ་ཆུང་དུ་ཞང་བཙན་བཙན་ཏོ་རེ་ལྷས་བྱིན་དང། སེང་གོ་སྣང་ཏོ་རེ་སྐྱི་ཟུང་གྱིས། བསྡུས་པར་ལོ་གཅིག ། ༆ ། སྟགི་ལོ་ལ་བབ་སྟེ། བཙན་པོ་དབྱརད་པོང་ཁྲི་མུ་སྟེངས་ན་བཞུགས། ཡུམ་ཁྲི་མ་ལོད་ཡར་འབྲོགི་འཇོན་ན་བཞུགས། དགུན་བཙན་པོ་ཁྲི་རྩེ་ན་བཞུགས། མདོ་སྨད་གྱི་དགུན་འདུན་ནམ་ལྡོང་པྲོམ་དུ་ཁུ་མང་པོ་རྗེ་ལྷ་ཟུང་དང། བློན་མང་རྩན་ལྡོང་ཞིས་བསྡུས་ཏེ། སུམ་རུའི་མཁོས་ཆེན་པོ་བགྱིས། ཡུམ་འོན་ཅང་དོ་ན་བཞུགས་ཅིང། འདུན་མའང་འོན་ཅང་དོར་བསྡུས་ཏེ། ཤིད་གྱི་བཀའ་ཏང་ཆེན་པོ་བོར་བར་ལོ་ཆིག ། ༆། ཡོས་བུའི་ལོ་ལ། བཙན་པོ་དབྱརད་གླིང་གི་འོལ་བྱག་ན་བཞུགས་ཤིང། རྒྱའི་ཕོ་ཉ་ཀམ་ཀེང་ཕྱག་འཚལད། མདོ་སྨད་གྱི་དགུན་འདུན་ཡོལ་གྱི་ཅི་འབོས་སུ་བསྡུས། ཡུམ་རྣང་ཕོ་དུར་མིག་ན་བཞུགས་ཤིང། ཁུ་འབྱུར་ལོད་བཙན་བཀུམ། དགུན་བཙན་པོ་འཇང་ཡུལ་དུ་གཤེགས་ཏེ། འཇང་ཕབ། པར་ལོ་གཅིག ། ༆ ། འབྲུགི་ལོ་ལ་བབ་སྟེ།
