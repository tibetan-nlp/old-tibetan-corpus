རུ་ལགི་ཞུགས་ལོང་དམར་པོ་བརྩིས། སེ་རིབ་གྱི་རྒྱལ་པོ་བཟུང། མདོ་སྨད་གྱི་དགུན་འདུན་གཙེ་ནམ་ཡོར་དུ་འདུས་པར་ལོ་གཅིག ། ༆ ། ཁྱི་ལོ་ལ་བབ་སྟེ། བཙན་པོ་བལ་པོ་ན་བཞུགས། པྱི་དྲོན་ན་བཞུགས། མཁྲིས་ཕ་ཏང་དུ་འདུན་མ་འདུས་ཏེ། བཙན་མོ་ཁོང་ཅོ་གཤེགས་པའི་ཡོ་བྱད་བཀྲལ། ཞང་བཙན་ཏོ་རེ་ལྷས་བྱིན་ལས་སོགས་པས། གཉེ་བོ་བགྱི་སྟེ། བཙན་མོ་ཀིམ་ཤང་ཁོང་ཅོ་ར་སའི་ཤ་ཙལ་དུ་གཤེགས། དགུན་བཙན་པོ་སྟངས་དབྱལ་བྲག་མར་ན་བཞུགས་པྱི་ལྷ་སྒང་ཙལ་ན་བཞུགས་ཏེ། མདོ་སྨད་གྱི་འདུན་མ་གཙེ་ནཾ་ཡོར་དུ་ཞང་རྒྱ་སྟོ་དང། སྟ་གུ་རི་ཙབ་གྱིས་བསྡུས་པར་ལོ་ཆིག ། ༆། པགི་ལོ་ལ། བཙན་པོ་དབྱརད་བལ་པོ་ན་བཞུགས། པྱི་དྲོན་ན་བཞུགས། དབྱར་འདུན་གཞོང་ཕྱག་དུ་འབོན་ད་རྒྱལ་བཙན་ཟུང་དང། བློན་ཆེན་པོ་ཁྲི་གཟིགས་གྱིས་བསྡུས་ཏེ། ཆོག་རོ་ཁོང་གེ་ལ་བཀྱོན་བབ། དགུན་བཙན་ཕོའི་པོ་བྲང་བྲག་མར་ན་བཞུགས། པྱི་ལྷས་གང་ཙལ་ན་བཞུགས། སྐྱི་རྣམས་སུ་འབོན་ད་རྒྱལ་བཙན་ཟུང་དང། བློན་བློན་ཆེན་པོ་ཁྲི་གཟིགས་གྱིས་བསྡུས་ཏེ། ཕ་ལོས་མང་པོ་བཀུག ། མདོ་སྨད་གྱི་འདུན་མ་གཙེ་ནམ་ཡོར་དུ་ཆོག་རོ་ཁྲི་གཟིགས། གནང་ཀོང་གྱིས་བསྡུས་པར་ལོ་གཅིག ། ༆། བྱི་བའི་ལོ་ལ། བཙན་པོ་དབྱརད་བལ་ཕོ་ན་བཞུགས། པྱི་དྲོན་ན་བཞུགས། འབོན་ད་རྒྱལ་དང། བློན་ཆེན་པོ་ཁྲི་གཟིགས་གྱིས། ལྷ་གབ་གྱི་བྱེ་མ་ལུང་དུ་བསྡུས་ཏེ། དགུན་བྲག་མར་ན་བཞུགས། བཙན་པོའི་མཚན་རྒྱལ་གཙུག་རུ་ལས། ཁྲི་ལྡེ་གཙུག་རྩན་དུ་གསོལད། པྱི་ཁྲི་མ་ལོད་ནོངས། དགུན་འདུན་སྐྱི་དྲ་ཙལ་དུ་འབོན་ད་རྒྱལ་དང། བློན་ཆེན་པོ་ཁྲི་གཟིགས་གྱིས་བསྡུས་ཏེ། རུ་གསུམ་གྱི་ཁྲམ་དམར་པོ་བརྩིས། ཕྱི་བཏོལ་བར་ལོ་གཅིག ། ༆། གླང་གྱི་ལོ་ལ་བབ་སྟེ། བཙན་པོ་དབྱརད་མལ་ཏྲོའི་བརྫེན་ཏང་ན་བཞུགས། འབོན་ད་རྒྱལ་དང་བློན་ཆེན་པོ་ཁྲི་གཟིགས་གྱིས། དབྱར་འདུན་ཟུ་སྤུགི་རྐྱང་བུ་ཙལ་དུ་བསྡུས་ཏེ། ལྔ་བརྒྱ་བསྐོས། སྔོ་ས་སྐྱ་ས། ཁ་བསྟནད་རྒྱའི་ཕོ་ཉ་ཡང་ཁེང་ཕྱག་འཚལད། དགུན་བཙན་ཕོའི་པོ་བྲང་བྲག་མར་ན་བཞུགས། པྱིང་བར་ཕྱི་ཁྲི་མ་ལོད་གྱི་མདད་བཏང། དགུན་འདུན་མོང་ཀར་དུ། བློན་ཆེན་པོ་ཁྲི་གཟིགས་གྱིས་བསྡུས་ཏེ། གནུབས་ཁྲི་མཉེན་མོན་ཙན་གུམ་ནས། ཁུད་པ་ཆེན་ཕོའི་འབངས་ཁྲི་སུམ་རྗེ་རྩང་བཞེར་ལ་ཕུལ་བར་ལོ་ཆིག ། ༆། སྟགི་ལོ་ལ་བབ་སྟེ། བཙན་པོ་དབྱརད་མལ་ཐྲོའི་བརྫེན་ཏང་ན་བཞུགས་ཤིང། འདུན་མ། མལ་ཐྲོའི་ལྟམས་སུ། བློན་ཆེན་པོ་ཁྲི་གཟིགས་གྱིས་བསྡུས་ཏེ། བྲུང་པ་རྡོ་འཕན་ཀོང་ནི་བྱུང། ཚེས་པོང་ཏྲེ་གོང་བཅུག་སྟེ། འབོན་ད་རྒྱལ་དང་ཞང་བཙན་ཏོ་རེ་ལྷས་བྱིན་གྱིས། སིལ་གུ་ཅིན་གྱི་འོ་ཁོལ་དུ་འ་ཞའི་མཁོས་བགྱིས། དགུན་བཙན་པོ་ཉེན་ཀར་ན་བཞུགས། དགུན་འདུན་མནོན་དུ་བློན་ཁྲི་སུམ་རྗེ་རྩང་བཞེར་གྱིས། བསྡུས་ཏེ། འབོན་ད་རྒྱལ་དང། བློན་ཆེན་པོ་ཁྲི་གཟིགས་གཉིས་གྱིས། འབུ་ཤིང་ཀུན་དུ་དྲ་མ་དྲངས་ཏེ། སླར་འཁོརད་པར་ལོ་གཅིག ། ༆ ། ཡོས་བུའི་ལོ་ལ། བཙན་པོ་ཞེ་ཤིང་གྱི་རྩིབས་ན་བཞུགས། དབྱར་འདུན་ཟུ་སྤུགི་རྐྱང་བུ་ཚལ་དུ་བློན་ཁྲི་སུམ་རྗེ་རྩང་བཞེར་གྱིས་བསྡུས་ཏེ། རྩང་ཅེན་གྱི་བྲུང་པ་ལང་ས་ཙེང་གུམ་སྟེ། ཞང་ཁྲི་མཉེས་སྨོན་ཟུང་བཅུག ། དགུན་བཙན་པོ་ཉ་ཤ་ཚལ་ན་བཞུགས། འདུན་མ་འོ་ཡུགི་ནུབས་སུ། བློན་ཁྲི་སུམ་རྗེ་རྩང་བཞེར་གྱིས་བསྡུས་ཏེ། བློན་ཆེན་པོ་ཁྲི་གཟིགས་གྱིས། མདོ་སྨད་གྱི་དགུན་འདུན་རྒྱོད་དུ་བསྡུས་ནས།
