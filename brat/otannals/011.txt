བློན་སྐྱེས་བཟང་ལྡོང་ཙབ་གྱིས། བྲུ་ཞ་ཡུལ་དུ་དྲངས། དགུན་པོ་བྲང་བྲག་མར་ན་བཞུགས་ཏེ། བྲུ་ཞའི་རྒྱལ་པོ་ཕབ་སྟེ་ཕྱག་འཚལད། རྒྱའི་ཕོ་ཉ་ཝང་འདོ་ཤི་ཕྱག་འཚལད། ནས། རྒྱས་ཆབ་སྲིད་བཤིག་པར་ལོ་གཅིག ། ༆ ། སྟག་གི་ལོ་ལ། བཙན་པོའི་པོ་བྲང་དབྱརད་དྲོན་ན་བཞུགས་ཏེ། ཆད་ཀ་བཅད། དགུན་པོ་བྲང་བྲག་མར་ན་བཞུགས། དགུན་འདུན་སྒྲེགས་གྱི་བྱ་ཙལ་དང་ཅུ་བགོའི་རྟེའུ་མཁར་དུ་འདུས་ཏེ། ཅད་ཀ་བརྩིས། སྐུན་ཀར་རྨ་ཚེ་སླར་ཐོབ་པར་ལོ་གཅིག ། ༆ ། ཡོས་བུའི་ལོ་ལ། བཙན་པོ་དབྱརད་ཆབ་སྲིད་ལ་བེག་དུ་གཤེགས་ཏེ། སྲས་ལྷས་བོན་དྲོན་ན་བཞུགས། བཞུགས་པ་ལས་ནོངས། བཙན་པོ་ཡབ་དགུན་བོད་ཡུལ་དུ་སླར་གཤེགས། བཙན་མོ་ཀིམ་ཤེང་ཁོང་ཅོ་ནོངས་པར་ལོ་གཅིག ། ༆ ། འབྲུག་གི་ལོ་ལ། བཙན་པོའི་པོ་བྲང། དབྱརད་མཚར་བུ་སྣའི་ངང་མོ་གླིང་ན་བཞུགས་ཏེ། ཇེ་བ་ཁྲི་མ་ལོད་བྲུ་ཞ་རྗེ་ལ་བག་མར་བཏང། དགུན་པོ་བྲང་བྲག་མར་ན་བཞུགས། ཞང་ཁྲི་མཉེས་སྨོན་ཟུང་གུམ་བར་ལོ་གཅིག ། ༆ ། སྤྲུལ་གྱི་ལོ་ལ། བཙན་པོ་དབྱརད་ཆབ་སྲིད་ལ་གཤེགས་ཏེ། རྒྱའི་མཁར་དར་ཁྭ་ཧྱྭན་ཕབ། ཞོ་དོན་གྱི་ཞང་ཙལ་དུ་བཙན་པོའི་སྤྱན་སྔར། ཁྲོམ་གྱི་མཁོས་ཆེན་པོ་བགྱིས། དགུན་ཆབ་སྲིད་ལས་སླར་བྲག་མར་དུ་གཤེགས་ཏེ། བཟོ་ཞལ་ཅོས་གྱིས། བརྒལད་ཏེ་མཁར་ལྕགས་རྩེ་སླར་ཐོབ། བཙན་པོ་སྲས་ལྷས་བོན་དང། བཙན་མོ་ཁོང་ཅོ་གཉིས་གྱི་མདད་བཏང། བར་ལོ་གཅིག ། ༆ ། རྟའི་ལོ་ལ། བཙན་པོའི་པོ་བྲང་དབྱརད་མཚར་བུ་སྣ་ན་བཞུགས། རྒྱའི་ཕོ་ཉ་ཨན་ད་ལང་དང། མྱྭ་ནག་པོའི་ཕོ་ཉ་ལ་བྲི་ཕྱག་འཚལད། ཟློར་ཤུད་པུ་ཁོང་ཟུང་དང། ལང་གྲོ་ཁོང་རྩན་གཉིས། འབྱུང་འཇུག་གི་རྩིས་བགྱིས་ཏེ། ཐང་ཁྲམ་བཏབ། ཁུ་ཉེ་མོན་གངས་སུ་བློན་མང་པོ་རྗེས་འ་ཞའི་མཁོས་བགྱིས། བཙན་པོ་སྲོང་ལྡེ་བརྩན་བྲག་མར་དུ། བལྟམ། ཡུམ་མང་མོ་རྗེ་ནོངས་པར་ལོ་གཅིག ། ༆ ། ལུག་གི་ལོ་ལ། བཙན་པོའི་པོ་བྲང་དབྱརད་ར་མཚར་ན་བཞུགས་ཏེ། བློན་ཆེན་པོ་ཆུང་བཟང་གྱིས། དབྱར་འདུན་བྲེང་དུ་བསྡུས་ཏེ། ཕ་ལོས་གྱི་བྱང་བུ་བོར། དགུན་པོ་བྲང་བྲག་མར་ན་བཞུགས། སྐྱི་རྣམས་སུ་རྒོད་གཡུང་གྱི་ཕ་ལོས་ཆེན་པོ་བཀུག ། བྲག་མར་དུ་རྒྱའི་ཕོ་ཉ་ཀྭག་ཅུང་ལང་ཕྱག་འཚལད་པར་ལོ་གཅིག ། ༆ ། སྤྲེའུ་ལོ་ལ། བཙན་པོ་དབྱརད་བྱང་རོལད་དུ་གཤེགས་པ་ལས་སླར་འཁོརད་ཏེ། པོ་བྲང་ར་མཚར་ཏུ་བཏབ། རྒྱའི་ཕོ་ཉ། ཅང་འགྭན་འགེ་དང། དུར་གྱིས་གྱི་ཕོ་ཉ་ཕྱག་འཚལད། ཡུལ་ཡུལ་དམག་མི་ཁྲམ་སྐྱ་བརྩིས། དགུན་པོ་བྲང་བྲག་མར་ན་བཞུགས། དགུན་འདུན་སྐྱི་ཤོ་མ་རར། བློན་ཆེན་པོ་ཅུང་བཟང་དང་བློན་སྐྱེས་བཟང་གཉིས་གྱིས་བསྡུས་ཏེ་དམག་མི་མཁོས་ཆེན་པོ་བགྱིས་ཏེ། བཙན་པོ་བཀས་ཁྲམ་དམར་པོ་ཤོག་ཤོག་སེར་པོ་ལ་སྤོས་པར་ལོ་གཅིག ། ༆ ། བྱ་གག་གི་ལོ་ལ། བཙན་པོ་དབྱརད་འོ་དང་ན་བཞུགས་ཤིང། དགུན་བྲག་མར་ན་བཞུགས། དགུན་འདུན་དྲ་བྱེར་འདུས། བྲུང་པ་ཞང་ཏྲེ་གོང་ཕྱུང་སྟེ། ཅོག་རོ་རྨ་གོང་བཅུག་པ་དང། སེང་གོ་འཕན་ལ་སྐྱེས་ཕྱུང་སྟེ། མྱང་འདུས་ཁོང་བཅུག་པའི་རྩིས་བགྱིས་ཏེ།
