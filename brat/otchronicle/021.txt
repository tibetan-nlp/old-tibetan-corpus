མལ་ཏྲོ་ནི་ཀླུམ་དང་ཉེ། སྐྱི་བསེར་ནི་སྤུ་རུ་རུ༎ ཇེ་ཉེ་ནི་ཇེ་ཉེ་ན། ར་ཉེ་ནི་ཤུག་པ་ཐུ༎ ཡར་གྱི་ནི་བྱེ་མ་ལ། ལྷན་ལྷན་ནི་འགྲོས་མོ་འདི༎ སྐྱི་འི་ནི་པུར་། ཟ་དུར་ནི་བཏབ་ཀྱང་རུང་ཞེས། མགུར་བླངས་པ་འི་རྗེས་ལ། རྨང་ཅུང་མཆིད་བྱག་བགྱིས་པ་ལས༎ བཙན་པོ་དྲལ་ལ། བཀའ་ལན་ཡི་གེར་གསོལ་བ་ནི་ཆང་མ་། བཙན་པོ་དྲལ་གྱི་ཞ་སྔ་ནས་ཐུགས་བདེ་ཞིང་བཞུགས་པ་གློ་བ་དགའ་ཞིང་མཆིས་སོ། བཙན་པོས་བཀའ་སྩལ་པ་ནི། གུམ་ཆད་གཉིས། རྔོ་ཇི་ཐོག་གིས་འཚལ་ཞིང་མཆིས་སོ༎ བཙན་པོ་འི་ཕྱག་དུ་འདི་པུལ་ཆིག་ཆེས། ཞུ་གཡག་ཞུ་གཅིག་ཕྱག་རྒྱས་བཏབ་ནས་བསྐུར་རོ༎ གྱིམ་བརྩན་རྨང་ཅུང་སླར༎ བཙན་པོ་སྤྱན་སྔར་མཆིས་ནས༎ བཙན་མོས་བཀའ་ལན་ཡི་གེར་གསོལ་པ་ནི་ཆང་མ་མཆིས༎ མགུར་བླངས་པ་འི་ཚིག་ནི་འདི་ཞེས་མཆི། ཕྱག་རྒྱས་བཏབ་སྟེ་སྐུར་བ་འདི་ལགས་ཤེས་གསོལ་པ་ལས༎ ཞུ་ཕྱག་རྒྱ་པྱེ་སྟེ། གཟིགས་ན། གཡུ་རྙིང་བཟང་པོ་སུམ་ཅུ་ཙམ་ཞིག་བྱུང་སྟེ༎ བཙན་པོ་དགོངས་དགོངས་ནས། ཡུ་བུས་ལིག་མི་རཧྱ་ལ་རྒོལ་ཕོད་ན་ནི་གཡུ་ཐོགས་ཤིག ༎ རྒོལ་མ་ཕོད་ན་ནི་བུད་མེད་དང་མཚུངས་ཞུ་གྱོན་ཆིག་ཟེར་བ་ལྟ་ཞེས་བཀའ་སྩལ་ནས༎ རྗེ་བློན་བློད་བློད་དེ༎ ལིག་མི་རཧྱ་འི་སྲིད་བརླག་གོ༎ ༆། ༔ ། རྒྱལ་འདི་འི་རིང་ལ༎ ཞང་ཞུང་གི་རྒྱལ་པོ་ལ་དྲངས་སྟེ། ཆབ་སྲིད་མཛད་ནས༎ ཞང་ཞུང་གི་རྒྱལ་པོ་ལིག་མི་རཧྱ་སྲིད་བརླག་སྟེ༎ ཞང་ཞུང་ཐམས་ཆད་འབངས་སུ་བཀུག་གོ༎ འུང་གི་རྗེས་ལ༎ བཙན་པོ་རྗེས་འབངས་དགྱེས་སྐྱེམས་སྟོན་མོ་གསོལ་ཏེ། བཙན་པོ་ཁྲི་སྲོང་བརྩན་གྱིས་མགུར་བླངས་པ༎ ཀྱེ་རྗེ་འི་ནི་མཚན་བ་འདི༎ ཁྲི་འི་ནི་སྲོང་བརྩན་ཞིག ༎ བློན་གྱི་ནི་མིང་བ་འདི༎ སྟོང་རྩན་ནི་ཡུལ་ཟུང་ཞིག ༎ ཆིབས་ཀྱི་ནི་མིང་བ་འདི༎ རྔུལ་བུ་ནི་གཙང་གཙང་ལྟ༎ གཙང་གཙང་ནི་ཡང་ཡང་ལྟ། ཡུ་བུ་ནི་རྗེ་དང་བློན༎ ཡར་མོ་ནི་ཆུ་ཐུངས་ཀྱིས༎ མདོ་ནས་ནི་རྩང་དུ་བསྲིང༎ ཡར་མོ་ནི་ཞེང་ཆུངས་ཀྱིས་ལྷོ་ནས་ནི་བྱང་དུ་བསྐྱེད༎ མ་བྲུན་ནི་མཐའ་བཞི་རྒྱལ༎ ཡུ་བུ་ནི་རྗེ་དང་བློན༎ རྗེ་འིས་ནི་འབངས་མ་གཏང། རྗེ་འིས་ནི་འབངས་བཏང་ན། གནམ་མཐའི་ནི་སྲུང་དུ་རུང༎ འབངས་ཀྱིས་ནི་རྗེ་མ་གཏང། འབངས་ཀྱིས་ནི་རྗེ་བཏང་ན།
