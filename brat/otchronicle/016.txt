འུང་ནས་ཞང་སྣང་གི་བྲན། པ་ཚབ་གྱིམ་པོ་འཁུས་ཏེ། ཞང་སྣང་བརླག་གོ༎ མཁར་སྡུར་བ་ཡང་བཤིག་གོ། ༆༎ འུང་གི་འི་འོག་དུ༎ ཁྱུང་པོ་སྤུང་སད་ཟུ་ཙེས༎ མྱང་ཞང་སྣང་བཙན་པོ་ལ་གློ་བ་རིངས་པ༎ ཟུ་ཙེས་དཀུའ་བེལ་ཏེ༎ ཞང་སྣང་བཀུམ་སྟེ༎ ཟུ་ཙེ་གློ་བ་ཉེ་འོ༎ བཙན་པོ་འི་བློན་པོ་འི་ནང་ན༎ ཁྱུང་པོ་སྤུང་སད་ཟུ་ཙེ་ལས་གློ་བ་ཉེ་བ་སྔོན་ཆད་མ་བྱུང་ངོ༎ ཟུ་ཙེ་འཛངས་ཤེ་ན་འཛངས༎ དཔའ་ཞེ་ན་དཔའ་འོ༎ ཆུ་གང་ཆེ་ཞེ་ན་ཆེ༎ ཡང་བ་རིང་ཞེ་ན་༎ མི་ཆིག་ལ་ཚོགས་དགུའ་ཚོགས་ན༎ སྤུང་སད་ཟུ་ཙེ་འོ༎ འུང་གི་འོག་དུ། བཙན་པོ་འདི་འི་རིང་ལ༎ ཁྱུང་པོ་སྤུང་སད་ཟུ་ཙེ་ལྟ་ཞིག ༎ རྒས་ཏེ་ཉི་མ་འདེར་གནང་ངོ༎ བཙན་པོ་ཞ་སྔར་ཁྱུང་པོ་སྤུང་སད་ཀྱིས་གསོལ་པ། སྔོན་བཙན་པོ་འི་ཡབ་གནམ་རི་འི་རིང་ལ༎ བདགིས་རྩང་བོད་འབངས་སུ་བཀུག་པ་ལྟ་ཞིག ༎ ཡབ་ཀྱིས་སྤྱན་གྱིས་མ་གཟིགས༎ ཞབས་ཀྱིས་མ་བཅགས་ན༎ བཙན་པོ་སྲས་ཀྱིས་སྤྱན་གྱིས་གཟིགས༎ ཞབས་ཀྱིས་བཅགས་ཏེ༎ སྡུམ་པ་ཁྲི་༎ དགྱེས་སྐྱེམས། སྟོན་མོ་གསོལ་དུ་ཇི་གནང་ཞེས་གསོལ་ནས༎ བཙན་པོས་སྤུང་སད་གསོལ་བ་བཞིན་གནང་སྟེ༎ མགར་ཡུལ་ཟུང་པོ་བྲང་ས་འདྲིར་བཀའ་སྩལ་ནས༎ ཡུལ་ཟུང་ཁྲི་བོམས་སུ་མཆིས་ཏེ། བརྟགས་ན༎ དཀུ་གང་པུབ་པར་ཡུལ་ཟུང་གིས་ཚོར་ནས༎ ཡུལ་ཟུང་སླར་བྲོས་ཏེ་བཙན་པོ་འི་སྙན་དུ་གསོལ་ཏོ༎ ཡུལ་ཟུང་སླར་བྲོས་པ་འི་རྗེས་ལ༎ ཁྱུང་པོ་སྤུང་སད་རང་ལྕེབས་ཏེ་གུམ་ནས། མགོ་བཅད་དེ༎ བུ་ངག་རེ་ཁྱུང་བསྐུར་ནས༎ སྐུར་མཁར་པྱིང་བར་མཆིས་སོ༎ བཙན་པོ་འི་སྙན་དུ་ངག་རེ་ཁྱུང་གིས་གསོལ་པ༎ ཕ་རྒས་ཀྱི། ལ་གར༎ བཙན་པོ་འི་ཞ་སྔར་གློ་བ་རིངས་ནས༎ ཚམས་བསྡལ་བར་ཆད་པ༎ མགར་ཡུལ་ཟུང་ལ་འཐོལ་བར་བགྱིས་པ་ལས། ཡུལ་ཟུང་གིས་ཀྱང་ཚོར་ཏེ། ཡུལ་ཟུང་ནི་སླར་མཆིས༎ བདགིས་ཕ་བཀུམ་ནས༎ མགོ་བཅད་དེ་མཆིས་ན༎ སྲིད་མི་བརླག་པར་ཇི་གནང་ཞེས་གསོལད་ཏོ༎ བཙན་པོ་ཁྲི་སྲོང་བརྩན་གྱིས་ཀྱང༎ ངག་རེ་ཁྱུང་གིས་གསོལ་པ་བཞིན་གནང་སྟེ༎ སྲིད་མ་། ༆། ༔ ། བཙན་པོ་ཁྲི་འདུས་སྲོང༎ སྐུ་ཆུང་ནས་གཞོན་གྱིས་ཀྱང༎ ཕག་རྒོད་ལ་བཤན་གྱིས་མཛད། གཡག་རྒོད་སྒོག་དུ་བཅུག ༎
