ཡུང་བ་སྣ་མན་ཆད༎ རྐོང་བྲེ་སྣར་ཡན་ཆད། བཙན་པོ་ཕྱག་དུ་བོན་ཏོ༎ བཙན་པོ་ཁྲི་སློན་བཙན་གྱིས། བཀའ་སྩལ་ཏེ། ཡུལ་ངས་པོ་ལས༎ འཕན་ཡུལ་དུ་མིང་སྤོས་སོ༎ མྱང་དབས་ཀྱིས༎ མཆིད་བླངས་པ༎ གཟུས་ནི་ལྷ་བོན་ཏོ། སློན་མཚན་སློན་ཀོལ། གཟུས་ནི་ལྷ་བོན་ཏོ༎ པྲིན་བྱ་རྒོད་ཀྱིས་བཏགས་ཏེ༎ ཚེས་པོང་ནག་སེང་གིས་བཏགས་སོ། རྗེ་བདེན་གྱིས་ནི་བཀོལ་ཏོ། སྒ་བདེན་གྱིས་ནི་བསྟད་དོ༎ མི་སྨོན་རུམ་དུ་བྱུང་ངོ༎ མཚེ་གཟུགས་རུམ་དུ་འཁྲུངས་སོ༎ དེ་ནས་ཡུལ་ངས་པོ་འི། འབངས༎ དབས་དབྱི་ཚབ་ལ་སོགས་པས༎ བཙན་པོ་འི་མཚན་གསོལ་པ༎ ཆབ་སྲིད་གནམ་བས་མཐོ༎ དབུ་རྨོག་རི་བས་བརྩན་ཏེ༎ གནམ་རི་སློན་མཚན་དུ་༎ མྱང་དབས་མནོན་དང་གསུམ་གྱིས༎ ཟིང་པོ་རྗེ་འི་སྲིད༎ བཙན་པོ་སྤུ་རྒྱལ་གྱི་ཕྱག་དུ་དངར་ཏོ༎ དེ་ནས་གནམ་རི་སློན་མཚན་གྱིས་ཕྱག་ལྕག་གིས༎ དྲས་ཏེ༎ མྱང་ཙེང་སྐུ་འི་བྱ་དགའར༎ མཉན་འཛི་ཟུང་གི་མཁར་སྡུར་བ་དང། བྲན་ཁྱིམ་སྟོང་ལྔ་བརྒྱའ་སྩལད་ཏོ། དབས་དབྱི་ཚབ་ཀྱི་བྱ་དགའར༎ ཟ་གད་གཤེན་ཀྱི་ཡུལ་ས་དང། མལ་ཏྲོ་པྱོགས་ནས་བྲན་ཁྱིམ་སྟོང་ལྔ་བརྒྱའ་སྩལ་ཏོ༎ མནོན་འདྲོན་པོའི་བྱ་དགའར། ཁོ་ན་འི་པུ་ནུ་པོ་མནོན་ལ་སོགས་པ་བྲན་ཁྱིམ་སྟོང་ལྔ་བརྒྱའ་སྩལ་ཏོ༎ ཚེས་པོང་ནག་སེང་གི་བྱ་དགའར། འོན་ཀྱི་སྨོན་མཁར་ནས༎ བྲན་ཁྱིམ་སུམ་བརྒྱའ་སྩལ་ཏོ༎ མྱང་ཙེང་ཅུང་དང། ཕ་སྤུན་པོ་མུ་གསེང་གཉིས་ནི་དཀུ་ལ༎ གཏོགས་སྟེ། དཀུ་རྒྱལ་པ་འི་ནང་དུ་ཡང་གཏོགས་སོ༎ དབས་དབྱི་ཚབ་ཀྱི་ཚ་བོ༎ སྟག་པོ་རྗེ་མེས་སྣང་དང། མང་པོ་རྗེ་པུ་ཚབ་གཉིས་དཀུའ་རྒྱལ་ལ་གཏོགས་སོ༎ ཚེས་པོང་ནག་སེང་གི་ནུ་བོ་ན་གུ་དཀུའ་རྒྱལ་ལ་གཏོགས་སོའ༎ འུང་ལྟ་སྟེ། མྱང་དབས་མཐོན་དང་གསུམ། ཚེས་པོང་པྲིན་དང་བཞིས༎ གློ་བ་ཉེ་ནས། བྲན་ཁྱིམ་མང་པོ་དང༎ ཡུལ་ཆེད་པོ་སྩལ་ཏོ༎ བཙན་པོ་འི་བློན་པོར་བཅུག་གོ༎ རྒྱལ་པོ་འདི་འི་རིང་ལ༎ ཁྱུང་པོ་སྤུང་སད་ཀྱིས། རྩང་བོད་ཀྱི་རྗོ་བོ་མར་མུན་མགོ་བཅད་དེ༎ རྩང་བོད་ཁྱིམ་ཉི་གྲི༎ བཙན་པོ་འི་ཕྱག་དུ་པུལད་ཏེ། ཟུ་ཙེ་གློ་བ་ཉེ་འོ༎ འུང་ནས། བཙན་པོ་སློན་བཙན་གྱིས༎ རྩང་བོད་ཁྱིམ་ཉི་གྲི༎
