བཙན་ཏེ་ནི་མི་མཁས་པ་རྒྱ་དྲུག་ནི་དགུང་མཐའ་རྗེ༎ མགྱོགས་ཏེ་ནི་མི་མཁས་པ། ཤུ་མ་ནི་རྒྱལ་མ་གཉིས། དྲག་སྟེ་ནི་མི་མཁས་པ་མཆོ་གར་ནི་འབྲོང་གི་རུ། རྣོ་སྟེ་ནི་མི་མཁས་པ་མདོ་ལྕགས་ནི་སྟེའུ་ཀ་མ། འཛངས་ཤིང་ནི་སྲི་ཞུ་བ། ཡར་ཀྱི་ལྷོ་རྔེགས་ཟའ། ངན་ཞིང་ནི་ཕོ་ཆེ་བ། འདམ་གྱི་ནི་ཅོག་རོ་ཟ། དྲི་ངན་ནི་ཉ་འི་དྲི། གཡའ་ངན་ནི་ཟངས་ཀྱི་གཡའ། སྐུ་འི་ནི་རིང་སྟོད་དུ། གཡང་གི་ནི་ཡུགས་བབ་བམ། སྐུ་འི་ནི་རིང་སྨད་དུ་ཚེར་ཀྱི་ནི་ཡུགས་ལྷུང་ངམ༎ སྔུན་ན་ནི་འདྲེན་པ་འདི། ཁྲི་གདས་ནི་ཡུལ་བྱིན་དང། མདོ་གྲགས་ནི་དགུང་དང་ཉེ༎ སླད་ནས་ནི་རྟོན་པ་འདི། རྒྱལ་ཚན་ནི་སྣང་གྲགས་དང། མུ་ཀེན་ནི་ཀླུ་ཅུ་རྨད༎ དབུས་ནས་ནི་འཕེན་པ་འདི༎ ཆེ་འམ་ནི་ཆུང་མ་འཚལ༎ འདུས་སྲོང་ནི་མང་པོ་རྗེ། སྐུ་ལ་ནི་འཕྲུལ་ཡང་མངའ༎ འོག་ན་ནི་བཅིབས་པ་འདི༎ རྒྱལ་མ་ནི་མཐའ་བཞི་འདུལ༎ མགྱོགས་སམ་ནི་བུལ་མ་འཚོ༎ རྨྱིག་རྗེས་ནི་མིས་མི་མཐོང༎ རྗེ་ཆིག་ནི་མང་པོ་རྗེ། གཤེགས་སོ་ནི་མང་པོ་རྗེ། བཞུགས་ཀྱང་ནི་འོད་མི་དྲོ། གཤེགས་ཀྱང་ནི་དང་མ་ཆགས༎ ཅོག་རོ་ཟས་མཆིད་བླངས་པ་འི་ཚིག་ལ༎ ཀྱེ་འདམ་གྱི་ནི་ཐང་ཀར་དུ། ལྷ་གཤེགས་ནི་ཞལ་མ་འཚོལ༎ ལྷ་གཤེགས་ནི་ཞལ་འཚོལ་ན། དུད་དེ་ནི་ཕྱག་ཀྱང་འཚལ༎ ཕྱག་མདའ་ནི་སླེབ་ཀྱིས་འཛིན༎ དེ་འི་ནི་མན་ཆད་ལ༎ དུད་དེ་ནི་ཕྱག་འཚལ་དུ། སོགས་བུ་ནི་ཡུ་བ་ཐུངས༎ འགྲེང་སྟེ་ནི་ཕྱག་འཚལ་དུ། སྟེན་ཀྱི་ནི་ཆོས་མ་ལགས༎ ༆། ༔ ། བོད་ཀྱི་བློན་པོ༎ མགར་ཁྲི་འབྲིང་བཙན་བྲོད་དང༎ རྒྱ་འི་བློན་པོ་ཝོང་ཀེར་ཞང་ཤེ་གཉིས༎ ཁ་དོར་གྱིས་བརྒྱའ་བ་འི་ཚིག ༎ རྒྱ་འི་དམག་བྱིན་ཆེད་པོ༎ དམག་པོན་ཝོང་ཀེར་ཞང་ཤེས་བདས་ཏེ༎ དཔུང་གིས་པུལ་ནས༎ མཚོ་སྔོན་པོ་པྱོགས་ཀྱི་དམག་པོན༎ མགར་ཁྲི་འབྲིང་བཙན་བྲོད་འཚལ་བ་ལ༎ ཝོང་ཀེར་ཞང་ཤེས་སྤྲིང་བ༎ ཁྲེ་རྐྱལ་གང་དང། ཡུངས་འབྲུ་རྐྱལ་གང་བསྐུར་ཏེ༎ བོད་ཀྱི་དམག ༎ སྟག་འཕྲང་གཡག་འཕྲང་དུ་བགྲངས་པ་འི་གྲངས་ཀྱང་ང་ལ་ཡོད་དོ༎ ། གླད་པ་ལ་ཚད་བླངས་སྟེ། ཞྭར་དྲུབས། རྐང་པ་ལ་ཚད་བླངས་ཏེ་ལྷམ་དུ་དྲུབས༎
