རྣ་བ་ལ་བཟུང་བ་ལ་སོགས་པ༎ ཐུགས་སྒམ་བ་འི་སྟེང་དུ། སྐུ་རྩལ་ཆེད་པོས་བསྣན་ཏེ། མི་དང་མི་འདྲའ་བར་འཕགས་པས༎ ཉི་ངོག་རྒྱལ་པོ་ཐམས་ཆད་དང། འབངས་མགོ་ནག་པྱོགས་ཀྱིས༎ མཚན་བླ་དགས་འཕྲུལ་གྱི་རྒྱལ་པོ་ཞེས། བཏགས་ཏེ། བཀའ་མཆིད་ཀྱི་དཔེར་བརྗོད་དོ༎ མགར་ལ་སོགས་པ་བློན་པོ་ཁ་ཆིག་གློ་བ་འདྲིང་ན༎ ཐུགས་སྒམ་བ་དང། དབུ་རྨོག་བརྩན་པོས་གཉེར་ནས༎ གློ་བ་རིངས་པ་ཐམས་ཆད་བཀྱོན་ཕབ་སྟེ༎ ཕྱག་ན་ཆབ་སྲིད་སྔོན་བས་ཀྱང། མཐོ༎ དྲུ་གུ་ལ་སོགས་སྟེ་ཉི་ངོག་གཞན་འབངས་སུ་བཀུག་ཅིང། དཔྱའ་ཕབ་པ་དང། མཁར་སྲ་བ་ཕབ། ཡུལ་པྱུག་པོ་བཅོམ་སྟེ་ཆབ་སྲིད་ལེགས་པ་མང་པོ་ནི་འདི་འི་ནང་དུ་མ་གཏོགས༎ འུང་གི་འོག་དུ་འཇང་ལ་ཆབ་སྲིད་མཛད་དེ་མྱྭ་དཀར་པོ་དཔྱའ་ཕབ༎ མྱྭ་ནག་པོ་བཀུག་པ་ལ་སོགས་ཏེ། རྒྱལ་པོ་འདི་ལས༎ དབུ་རྨོག་བརྩན་ཞིང། ཆབ་སྲིད་མཐའ་སྐྱེས་པ་ནི་སྔན་ཆད་གདུང་རབས་ཀྱིས་མ་མཛད་དོ༎ ༆༎ རྒྱལ་པོ་ཁྲི་ལྡེ་གཙུག་བརྩན་གྱི་རིང་ལ༎ རྗེད་གཤིན་ཆིས་འཇམ་སྟེ། མི་ཡོངས་ཀྱིས་སྐྱིད་དོ༎ བློན་ཆེ་སྟག་སྒྲ་ཁོང་ལོད་དང། རྗེ་བློན་ནས༎ རྒྱལ་པོ་ཞབས་ཀྱིས་བཙུགས་ཏེ༎ རྒྱ་ལ་ཆབ་སྲིད་མཛད་ན༎ རྒྱ་འི་མཁར་ཀྭ་ཅུ་ལ་སོགས་པ་ཕབ་སྟེ༎ དེ་ཙམ་ན་རྒྱ་འི་སྲིད་ཆེས་ནས༎ བྱང་ཕྱོགས་ཀྱི་དྲུ་གུ་ཀུན་ཀྱང་འདུས་ལ༎ ཏ་ཟིག་ལ་ཐུག་པ་མན་ཆད་རྒྱ་འི་ཁམས་སུ་གཏོགས་སྟེ༎ རྒྱ་འི་ནོར་མང་པོ་སྟོད་པྱོགས་སུ་འདོན་པ་རྣམས༎ ཀྭ་ཅུ་ན་ཚོགས་བྱས་པ་ལས༎ ཐམས་ཆད་བོད་ཀྱིས་ཕབ་སྟེ་བཞེས་པས༎ བླར་ཡང་དཀོར་མང་པོ་བརྙེས༎ འབངས་མགོ་ནག་པོས་ཀྱང། རྒྱ་དར་བཟང་པོ་ཁྱབ་པར་ཐོབ་བོ༎ ལྷོ་པྱོགས་ཀྱི་སྨད་ན་འཇང་དུམ་མྱྭ་དཀར་པོ་ཞེས་བྱ་བ་འི་རྒྱལ་པོ་སྡེ་མི་ཅུང་བ་ཞིག་འདུག་པ༎ རྒྱལ་པོ་ཐུགས་སྒམ་པོ་འི་རླབས་དང་ཐབས་ཀྱིས་བཀའ་སྩལ་ཏེ། མྱྭ་འི་རྒྱལ་པོ་ཀག་ལ་བོང་ཞེས་བྱ་བ༎ འབངས་སུ་ཕྱག་འཚལ་ནས། ཐབས་གཅུང་སྩལ་ཏེ། མི་མང་གི་སྣོན་བཏབ། ཡུལ་ཆེ་འི་ནི་འདབ་བསྐྱེད་དོ༎ འཇང་གི་རྒྱལ་པོ་བོད་ཀྱི་འབངས་སུ་བཞེས་པས༎ རྒྱ་འི་ཆབ་སྲིད་ལ་ཡང་ཤིན་དུ་དམའ་ཞིང། མི་བདེ་བར་བྱས་སའོ༎ འཇང་མྱྭ་འི་རྒྱལ་པོ་ལྟ་ཞིག་རྒྱ་ལ་ལྟ་ལྟ་བ་ལས༎ རྒྱ་རྗེས་དགྲར་བླངས་ཏེ༎ བཙན་པོ་ཁྲི་ལྡེ་བཙུག་བརྩན་གྱི་ཞ་སྔར་གློ་བ་ཉེ་ནས༎ རྒྱ་ཡུལ་ཕབ་ཀྱིས་ནི་ཡུལ་དང༎ མཁར་དུ་བཅས་ཏེ་ཕུལ༎ གཡུལ་སྤྲད་ཀྱིས་ནི། རྒྱ་བཙོན་ལུག་ལྟར་བསྐྱངས་སོའ༎ འུང་ནས་བཙན་པོ་ཁྲི་ལྡེ་གཙུག་བརྩན་གྱི་ཞ་སྔར༎ ཕང་དང་གི་ཁང་མོ་ཆེར༎ ཀག་ལ་བོང་གི་བློན་པོ། དྭན་ཅུང་ཀོག ། ཕྱག་འཚལ་བ་འི་ཚེ༎ བཙན་པོ་རྗེས་འབངས་ཀྱིས་མགུར་བླངས་པ་འི་ཚིག་ལ༎ དགུང་སྔོ་ནི་བདུན་རིམ་གྱི༎ ལྷ་ཡུལ་ནི་གུང་དང་ནས༎ ལྷ་སྲས་ནི་མི་འི་མགོན། མི་ཡུལ་ནི་ཐམས་ཆད་དང། མི་མཚུངས་ནི་མི་འདྲའ་སྟེ༎ ཡུལ་མཐོ་ནི་ས་གཙང་བས༎ བོད་ཡུལ་ནི་གཤང་དུ་གཤེགས༎ མི་ཡུལ་ནི་ཀུན་ཀྱི་རྒྱལ༎ ཆོས་བཟང་ནི་གཙུག་ཆེ་བས༎ རྒྱལ་པྲན་ནི་ཀུན་ཀྱང་འདུམ༎ ན་ནིང་ནི་གཞེ་ནིང་སྔ། ཆབ་ཆབ་ནི་མ་གཞུག་ནས༎ ལ་བོང་ནི་རྗེ་དང་སྐོལ། བློ་ཆེ་ནི་དཀྱེལ་མཁས་ལ༎ རྒལ་མཁས་ནི། ཁོང་དཔའ་བས༎ རྒྱ་རྗེ་ནི་བསམ་ལང་ལ། དགྲའ་བཞེར་ནི་གཡུལ་སྤྲད་ཅིང། དགྲ་ཟིན་ནི་གསར་སྤངས་པས༎ སྲིད་ཀྱི་ནི་མགོ་བཟུང་ཞིང། ཕ་སྐྱབས་ནི་སྡུག་བཙལ་པ༎ ལྷ་སྲས་ནི་བཙན་ལ། བཙལ༎ ལྷ་སྲས་ནི་གཙུག་ཆེ་ལ༎ ཆོས་བཟང་ནི་རྗེད་གཤིན་ཞིང་བཀའ་དྲང་ནི་གསུང་རྟག་ལ། ལ་བོང་ནི་སྲིད་གསོལ་བ། མི་སྲིད་ནི་ལྷས་མཛད་ཀྱིས། སྲིད་ཆེ་ནི་མཐར་མི་འགྱུར། ལ་བོང་ནི་རྗེ་དང་སྐོལ། དཔའ་འི་ནི་ལས་བྱས་པ༎ རྒྱ་མཁར་ནི་མཐོའ་བཤིག་ཅིང། རྒྱ་འབངས་ནི་མང་བསྡུས་ནས། ཡུལ་དང་ནི་སྡེར་བཅས་ཏེ། བོད་ཡུལ་ནི་ཐིལ་དུ་བགྱིས། ཡར་ཏེ་ནི་གནམ་ཡང་དགྱེས། མར་ཏེ་ནི་ས་ཡང་དགའ། ལ་བོང་ནི་རྗེ་དང་སྐོལ། ཇེ་ཉེ་ནི་ཇེ་ཉེ་འོ། ལྷ་ཉེ་ནི་གུང་དང་ཉེ། ཇེ་བརྟན་ནི་ཇེ་བརྟན་ན། ཤམ་པོ་ནི་གཉན་ལ་བརྟན། དི་རིང་ནི་སང་ལྟ་ན། ཅུང་ཀོག་ནི་དཔོན་དང་གཡོག ། གནམ་ས་ནི་ཡིད་སྦྱོར་བ། ན་བུན་ནི་ལྷ་འི་སྦུབས། འགེང་གང་ནི་ཀུན་ལ་དཔེན༎ སྔ་བསུས་ནི་པྱིས་བསྐྱལ་ཞིང། ཀླུ་བླངས་ནི་བྲོ་བརྡུངས་པ། དགོས་ཀྱིས་ནི་རིམ་གྲོ་བགྱིའ༎ ༆། ༔ ། བཙན་པོ་ཁྲི་སྲོང་ལྡེ་བརྩན་གྱི་རིང་ལ༎ ཆོས་བཟང་སྲིད་ཆེ་སྟེ། རྒྱལ་པོ་ནི་གནམ་ས་གཉིས་ཀྱི་བར་ཡུལ་དུ་བརྣམ་ཞིང༎ འགྲེང་དུད་གཉིས་ཀྱི་རྗེ་དང་བདག་མཛད་པ་འི་གཙུག་ལག་ཆེན་པོ། མི་འི་དཔེར་རུང་བར་མཛད་དའོ༎ ལེགས་ཀྱི་བྱ་དགའ་ནི་རངས་པར་བྱིན། ཉེས་ཀྱི་ཆད་པ་ནི་དམིགས་སུ་ཕོག་པར་མཛད་དོ༎
