གསོལ་དགུང་ཡང་གནང་སྟེ༎ བཙན་པོ་ཞ་སྔ་ནས༎ བཀའ་སྩལད་པ༎ མི་དྲིན་[]་མི་ནི་སྙིང་མི་གཅད༎ རྟ་ལྟེ་འུ་རྟ་ནི་སྒལ་མི་དབྱུང༎ སྔ་ན་ང་འི་ཡབ་ཀྱི་རིང་ལ༎ འབྲིང་ཐོ་རེ་སྦུང་བརྩན། དང༎ ཕངས་ཏོ་རེ་དབྱི་ཚབ་དང་ཁྱེད་གཉིས༎ ཆབ་ཕ་རོལ་ཚུ་རོལ་གྱི༎ གུང་བློན་འཚོལ་ཆིག་པར་དགོད། དགོད་པ་ལས། འབྲིང་ཏོ་རེ་སྦུང་བརྩན་ནི་ཤི༎ ཁྱོད་ནི་རྒས་སྟེ་ཉི་མ་འདེར་གནང་སྟེ་བཞག ༎ མྱང་ཞང་སྣང་བློན་ཆེར་བསྐོས་ཏེ་བཞག་ན༎ ཡབ་ནོངས་ནས། མྱང་གིས་ཀྱང་གློ་བ་རིངས་ཏེ། བཀྱོན་ཕབ་པ། ཡིན་ནོ། ད་ཁྱེད་དབས་ཕ་ཆིག་ནི༎ གདོའ་ཐག་གཉིས་སུ། གློ་བ་མ་རིངས་པས༎ ལ་མོ་ཆག་པ་པྲུམ་དུ་ཕྱག་ཐབ་ཀྱང་གསོལ་ཆིག ༎ དབུ་སྙུང་ཡང་གནང་ངོ་ཞེས་བཀའ་སྩལ་ཏོ༎ ལ་མོ་ཆག་པ་པྲུམ་དུ༎ རྩང་པྱེད་པུབ་སྟེ༎ ཕྱག་ཐབ་གསོལ་ཏོ༎ སྐུ་རྟེན་དུ་ཁྲབ་བསེའ་སྣ་བཅུ་དང། ལྡོང་པྲོམ་གྱི་རལ་གྱི་མདོར་ཅོད། གཉིས་གསོལ་ཏོ༎ བཙན་པོས་དབུ་སྙུང་ཡང་གནང་ངོ༎ དབས་དབྱི་ཚབ་སྤུན་མཚན་བདུན་ཡང་བྲོ་སྩལ་ཏོ༎ བཙན་པོས་བཀའ་སྩལད་པ༎ དབྱི་ཚབ་གློ་བ་ཉེ་བས༎ གུམ་ན་མཆད་ཕྱག་དར་ཏེ་བརྩིག་པར་གནང། རྟ་ནི་བརྒྱའ་དགུམ་བར་གནང༎ བུ་ཚ་གང་རུང་བ་གཅིག ༎ གསེར་གྱི་ཡི་གེ་མི་ཆད་པར་སྩལད་པར་བཀའ། སྩལ་ཏོ༎ བཙན་པོ་ཞ་སྔ་ནས༎ མགུར་བླངས་པ༎ ཡར་མོ་ནི་ཆུ་ཐུངས་ཀྱིས༎ མདོ་ནས་ནི་རྩང་དུ། བསྲིང། ཡར་མོ་ནི་ཞེང་ཆུངས་ཀྱིས། ལྷོ་ནས་ནི་བྱང་དུ་བསྐྱེད༎ འཐམ་འཐམ་ནི་འདུ་འདུ་ན། ངག་རྗེས་ནི་མིས་མི་བརྗོད༎ འགྲོ་འགྲོ་ནི་འཆམ་འཆམ་ན༎ ཆགས་ལྷམ་ནི་ཆུས་མི་སྣང། ད་ནས་ནི་ཕན་ཆད་དུ། ཁྱོད་ཀྱིས་ནི་ང་མ་གཏང། ང་འིས་ནི་ཁྱོད་མི་གཏང༎ ང་འིས་ནི་ཁྱོད་བཏང་ན༎ དགུང་མཐའི་ནི་སྲུང་དུ་རུང༎ ཁྱོད་ཀྱིས་ནི་ང་བཏང་ན༎ ངས་པོ་ནི་རྨད་དུ་རུང་ཞེས་བཀའ་སྩལ་ཏོ༎ དེ་ནས་དབས་དབྱི་ཚབ་ཀྱིས་མཆིད་བླངས་པ། བཙན་དུ་ནི་བདག་འཚལ་ན། ཆུང་ན་ནི་རྗེ་ཁུར་ཆིག ༎ པྱུག་དུ་ནི་བདག་འཚལ་ན༎ ལྷ་རི་ནི་གཡང་སྐྱོང་ཤིག ༎ རྗེ་འིས་ནི་བཀའ་སྩལ་ན་གཞའ་མ་ནི་ཡུན་ཀྱི་སྲིད༎ བྱ་འིས་ནི་ཞུ་པུབ་ན་ལ་ཕྱི་ནི་གདངས་སུ་དྲོ༎
