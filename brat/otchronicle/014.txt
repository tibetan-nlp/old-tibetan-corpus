ཞེས་མཆིད་བླངས་ནས༎ སྤད་མཚན་བདུན་གྱིས་གཏང་རག་ཆེད་པོ་བཏང་ངོ༎ དབུ་སྙུང་གནང་བ་འི་གོར་མ་ཕྱག་དུ་བོན་པ་ནི༎ རྒྱལ་འབྲིང་ལན་ཏོན་གྱིས་བླངས་ཏེ་ཕྱག་དུ་གསོལ་ནས། གོར་མ་དཀར་པོ་དངོས་ནི༎ དབྱི་ཚབ་ཀྱི་མཆད་པ་འི་རྨང་དུ་༎ ཞང་ལོན་ཞ་འབྲིང་དུ་བྲོ་འདོར་འདོར་བ། ཁུ་ཁྲི་དོ་རེ་སྨྱང་ཟུང་དང། གནུབས་སྙ་དོ་རེ་གཙུག་བློན་དང། རྔེགས་རྒྱལ་འབྲིང་ལན་ཏོན་དང། ཚེས་པོང་ཁྲི་བཙན་ཁོང་སྟོ། དང། འོ་མ་ལྡེ་ཁྲི་བཟང་ལོད་བཙན་དང། ཁུ་སྨོན་ཏོ་རེ་ཕངས་ཚབ་དང། རྗེ་བློན་བདུན་གྱིས་དེ་ལྟར་དབུ་སྙུང། གནང་ངོ༎ དབུ་སྙུང་གནང་བ་འི་ཚིག་ནི། ད་པྱིན་ཆད་ཁྱེད་སྤད་མཚན་གྱིས༎ བཙན་པོ་སྤུ་རྒྱལ་གྱི་ཞ། སྔར། གློ་བ་མ་རིངས་མཐའང་མ་གྲང་ན༎ ནམ་ནམ་ནམ་ནམ། ཞ་ཞ་ཞ་ཞར་ཡང། དབྱི་ཚབ་གྱི་བུ་ལ་མ་ཉེས་པར་བཀྱོན་དབབ་རེ། འཕྲའ་མ་མི་དག་གསན་རེ། འཕྲའ་མ་ཞིག་གསན་ན་ཡང། ཤགས་བྱ་སྟེ། ཡུས་བཀྲོལ་ནས༎ བཀས་མི་གཅད་རེ༎ དབྱི་ཚབ་ཀྱི་བུ་ཚ་ལ་ལ་ཞིག་གིས། མ་བསམས་སྟེ་སྙིང་རིངས་ན་ཡང༎ གང་སྙིང་རིངས་པ་ཉི་ཙེ་འི་སྒོར་མི་བཀྱོན་རེ༎ གཞན་བློ་ལ་མ་གཏོགས་པ་རྣམས་ལ། བཀྱོན་རེ། ཕག་དང་མཚུངས་མཚོན་གྱིས་མི་དགར་རེ། བྱ་དང་མཚུངས་ཁྱུས་བདའ་རེ། མ་ཉེས་པ་ཞིག་གིས་རྔོ་ཐོག་ན༎ བསྐྱུང་རེ་སྨད་རེ༎ དབྱི་ཚབ་ཀྱི་བུ་གཅིག་ལ། གསེར་གྱི་ཡི་གེ་ལས་སྨད་རེའ༎ ཁོལ་ཡུལ་ཡང་མ་ཉེས་པར་དཔྲོག་རེ༎ སྙིང་མ་རིངས་པར་རབས་ཆད་ན། ཁོལ་ཡུལ་བཞེས་རེའ༎ ཁྱེད་གློ་བ་རིངས་ན་དབུ་སྙུང་ལ་མ་གཏོགས༎ ཁྱེད་ཀྱིས་ཁོལ་ཡུལ་ཞིག་བརྗེས་སམ་པུལ་ན། དབུ་སྙུང་ལ་མ་གཏོགས། ནམ་ནམ་ནམ་ནམ། ཞ་ཞ་ཞ་ཞར། སྲས་དབོན་གྱི་ཞ་སྔར་འདི་བཞིན་དུ་གནང་ངོ༎ དབྱི་ཚབ་སྤད་སྤུན་བྲོ་སྩལད་པ། བཙན་པོ་སྤུ་རྒྱལ་ཁྲི་སྲོང་བརྩན་ཡབ་སྲས་དང། གདུང་རྒྱུད་ལ། གློ་བ་འདྲིང་རེ། ནམ་ནམ་ཞ་ཞར་ཡང༎ དབོན་སྲས་རྒྱལ་པོ་གང་མཛད་པ་ལ། གློ་བ་འདྲིང་རེ༎ གཞན་མི་འོ་ཆོག་ལ་རྗོ་བོ་ལྷག་པ་ཚོལ་རེ༎ གཞན་གློ་བ་འདྲིང་བར་སེམས་པ་དང་བློད་རེ། ཟས་ཀྱི་དོན་བཞག་པ་དང། དུག་དུ་བསྲེ་རེ། བཙན་པོ་ཁྲི་སློན་བཙན་ལ།
