ཆོས་ནི་ཆོས་སུ་མ་རུང་སྟེ། ཡུལ་གྱི་མི་དབུལ། ཉམས་ཀྱི་སྲིད་དམའ་སྟེ། ན་མོ་བརླག་ཟིན་ན་ཇི་ནོངས་ཤེས་གསོལ་ནས། འུང་ནས། ཟིང་པོ་རྗེ་སྟག་སྐྱ་བོ་ཉན་དུ་མ་རུང་སྟེ༎ འདི་སྐད་ཟེར་བ་ཉེས་སོ་ཞེས༎འཛི་ཟུང་བློན་པོ་ལས། པྱུང་ངོ༎འུང་ནས་འཛི་ཟུང་མ་རངས་སྟེ༎ ཟིང་པོ་རྗེ་ཁྲི་པང་སུམ་ལ་ལྟས་ནས༎ ནག་པོ་འཁུས་ཏེ། སྐྱ་བོ་བསད། དྲེ་འུ་རྒལ་ཏེ་བསེའ་སྒ་བཅག་གོ༎སྟག་སྐྱ་བོ་འི་ཡུལ། ཡུལ་ཡེལ་རབ་སྡེ་བཞི་དང༎ ཀླུམ་ཡ་གསུམ༎ ཟིང་པོ་རྗེ་ཁྲི་པང་སུམ་གྱིས། འདུས་སོའ༎དེ་འི་དཀུའ་འགེལ་དུ་མཁར་སྡུར་བས་བཅད་དེ། ཀླུམ་ཡ་སུམ་གྱི་སྨད༎མཉན་འཛི་ཟུང་གི་བྲན་དུ་དངར་ཏོ༎ བྲན་གྱི་ས་རིས་ཀྱི་ནང་དུ༎ མྱང་ནམ་ཏོ་རེ་ཁྲུ་གྲུ་དང། སྨོན་ཏོ་རེ་ཙེང་སྐུ་སྤད་ཀྱང༎འཛི་ཟུང་གི་བྲན་དུ་བཅད་དོ༎ འུང་ནས། འཛི་ཟུང་གི་ཆུང་མ་པ་ཙབ་ཟས། མྱང་བྲན་བྱེད་པ་ལ༎ དྲེགས་བཀུར་ཏེ༎ སྡིག་བསྣོལ། སྟུ་བསྟན། ནས། མྱང་ཙེང་སྐུ་མ་རངས་ཏེ༎ ཟིང་པོ་རྗེ་ཁྲི་པང་སུམ་ལ༎མཉན་གྱི་བྲན་བྱེད་མི་ཕོད་ཆེས་འཁངས། སྟེ་གསོལད་ཏོ༎ ཟིང་པོ་རྗེ་ན་རེ༎ མཉན་འཛི་ཟུང་ལས་ང་ལ་སྙིང་ཉེ་མེད་པ་འི་བྲན་དུ༎ ཁྱོན་འདའ་མ་རན་ལ་མེད་དོ༎ རྗོ་མོས་སྡིག་བསྣོལ་སྟུ་བསྟན་བ་བས༎ ཁར་གླན་ཡང་རྗོ་མོ་དབང་གིས། ཉེས་པ་མེད་དོ་ཞེས་སྟེ། གསོལ་བ་མ་གནང་ངོ༎འུང་གི་ཚིག་དེས་ཙེང་སྐུ་ཤིན་དུ་མ་རངས་སྟེ་ཡི་ཆད་དོའ༎འུང་གི་རྗེས་ལ༎ཟིང་པོ་རྗེ་འི་མངན་དབས་བཤོས་ཏོ་རེ་ཁྲུ་གུ་དང༎གཤེན་ཁྲི་བཞེར་འདྲོན་ཀོང་གཉིས། འཕྲེན་བ་འི་རྫིང་ཀར་འཐབས་ནས༎ གཤེན་ཀྱིས་དབས་བསད་དོ༎ དབས་ཀྱི་ཕོ་བོ་ཕངས་ཏོ་རེ་དབྱི་ཚབ་ཀྱིས། ཟིང་པོ་རྗེ་ལ། ནུ་བོ་གཤེན་ཀྱིས་བསད་པ་འི་སྟོང་ཇི་གནང་ཞེས་གསོལ་ནས༎ ཟིང་པོ་རྗེས༎གཤེན་ཁྲི་བཞེར། ཁོལ་ཚབ་ནང་བློན་བྱེད་པའི་ངོ་མ་ཆོད་དེ༎ དགེ་བས་མི་དགེ་བ་བསད་ན། བསད་དུ་ཟད་དོ་ཞེས་འོ༎ འུང་གི་ཚིག་དེས༎ དབྱི་ཚབ་ཤིན་དུ་མ་རངས་སྟེ་ཡི་ཆད་དོ༎ འུང་ནས་མྱང་དབས་གཉི་ཡི་ཆད་དེ། འགྲོགས། སྟེ་དོང་ངོ༎དབྱི་ཚབ་ནི་སྔ། ཙེང་སྐུ་ནི་འཕྱི་སྟེ༎
