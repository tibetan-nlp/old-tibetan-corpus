ཚང་མི་ཞིག་པར་སྒོ་ང་མི་ཆག་སྟེ༎ རིས་སྒོ་ང་གཅག་དུ་དེ་ལྟར་མི་ཆོགས་སོ༎ མེ་རི་ཀ་ན་འབར་བ། ཆུ་ལུང་པ་ན་འབབ་པས་གསད་དུ་མི་སླེབས་ཏེ༎ བོད་ཀྱི་སྤུ་རྒྱལ་ནི་ཉི་མ་དང་འདྲའ༎ རྒྱ་རྗེ་ནི་ཟླ་བ་དང་འདྲ་སྟེ༎ རྒྱལ་པོ་ཆེད་པོར་འདྲའ་མོད་ཀྱོང། གནམ་མཐའ་འོག་དུ་འོད་ཐ་དད་དོ༎ ཆེ་ཅུང་གི་ཁར་ཡོང་མི་དོར་རོ༎ མཚོ་ཆེད་པོ་འི་ནང་ན། ཆུ་སྲིན་རྒྱལ་བ་འདུག་ན། གནམ་ལས་ཆེ་ཕབ་སྟེ༎ ཆུ་སྲིན་རྒྱལ་བ་ཆུ་འི་ནང་དུ་བསད་དོ༎ འབྲུག་དང་ཀླག་གཉིས་ཀྱིས་ལྕེ་ཕབ་ནས་བྲག་ལས་སྲ་བ་མེད་པ་ཡང་དྲལ་ལོ༎  བོད་ཀྱི་ལྷ་བཙན་པོ་དང། གནམ་གཉིས་ཀྱིས་བདུད་བཅད་ན༎ འབལ་ལྗི་རྨང་ལས་ཆེ་བ་མེད་པ་འི་བུ༎ ས་དགུའ་རིམ་འོག་དུ་སྦས་པ་ཡང་བཟུང་དེ་བསད་དོ༎ དེ་ལྟ་སྟེ། ཆེ་ཅུང་དང་མང་ཉུང་གི་ཁར་དེ་ལྟར་མི་དོར་རོ། དེ་ནས་རྒྱ་འི་དམག ༎ ཝེང་ཀེར་ཞང་ཤེས་བདས་ཏེ་བྱུང་བ་ལས༎ བོད་ཀྱི་དམག་པོན། བློན་ཁྲི་འབྲིང་གིས། དགྲ་ཐབས། []་དང་གཡག་ལྟར་བྱས་ནས༎ གོ་བར་དུ་གཡུལ་སྤྲད་དེ༎ རྒྱ་མང་པོ་བཏུངས་ནས། རྒྱ་འི་རོ་གཅིག་གནམ་དུ་འགྲེང་བ་ཡང་མི་འབུམ་བསད་པ་འི་མཚན་མ་ཞེས་བྱ་འོ།  སྟག་ལ་རྒྱ་དུར་དང་རྨ་རྒྱ་དུར་ཀྱང་དེ་ནས་། རྔ་རབ་ཕན་ཆད༎ མྱང་ད[]་[]། དོར་པོ་ལ་སོགས་པ༎ རྒྱལ་པོ་དང་བཅས་སུ་བཀུག་ནས༎ སོ་བློན་སྡེ་ལྔ་ཡང་བཙུགས་སོ༎ []ང་ཀོག་ལ་སོགས་པ༎ སྟོད་པྱོགས་དང་ལྷོ་པྱོགས་ཀྱི་རྒྱལ་པྲན་མང་པོ་ཡང་བསྡུས་སའོ༎ []་མང་པོ་རྗེ་སྟག་[]་ཚབ་རྒྱལ་ཏོ་རེ་རྒྱ་ལ་བྲོས་པ་འི་ཚེ༎ པ་ཚབ་རྒྱལ་ཏོ་རེས་ཀླུ་བླངས་པ༎ []་ན་ལས༎ སྨན་ཤ་ནི་ལ་འདས་ཀྱང། ཤ་བཀྲའ་ནི་སླད་ན་ལུས། ན་ནིང་ནི་[]་འགྲེས་སུ་ནི་སྤང་བཟངས་ནས། མ་སྲིང་ནི་སྡུག་གིས་བསྐྱལ། ཙོང་ཀ་ནི་ཆེ་ཅུང་[]་རྗེ་འི་ནི་སྐལ་ཕོག་པ་རྒྱ་རྗེ་ནི་བསམ་ལང་ཞིག ། པུར་མི་ནི་སྐལ་བ་རུ་རྒྱ་[]་ཆག་པ༎ མགར་མང་པོ་རྗེ་སྟག་རྩན་གྱི་མཆིས་བྲང༎ ཅོག་རོ་ཟ་ཁྱོ་དང་ལྷན་ཆིག་རྒྱ་ལ་[]་མཆིད་བླངས་པ༎  ཉེན་ཀར་ནི་སྟག་རྩེ་ནས། ལྕགས་ཀྱི་ནི་ཁྱི་བཀྱེ་ཞིང། རྨ་འི་ནི་པོང་བྲ་ནས་[]་དུད་བརྡའ་བཏང། མེག་ལེ་ནི་གླང་མར་ནས།
