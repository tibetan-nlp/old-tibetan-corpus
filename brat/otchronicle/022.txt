ངས་པོ་ནི་རྨད་དུ་རུང༎ མགུར་དེ་ཞེས་གསུངས་སོ༎ མགུར་གྱི་ལན༎ བློན་སྟོང་རྩན་གྱིས་གསོལ་པ། རྗེ་འིས་ནི་བཀའ་སྩལ་པ་གཞའ་ཕྱི་ནི་ཡུན་ཀྱི་སྲིད། བྱ་འིས་ནི་ཞུ་པུབ་པ། ལ་ཕྱི་ནི་དྲོ༎ གློ་བོ་ནི་མང་ཀུ་རྒྱ། ནམས་ཀྱང་ནི་ཡར་མི་པོ། གད་ཀྱི་ནི་གསེར་པོ་ཆེ༎ ནམས་ཀྱང་ནི་པྱིང་བ་འི་བཅུད། རྗེ་འིས་ནི་འབངས་མ་གཏང༎ རྗེ་འིས་ནི་འབངས་བཏང་ན། བྱ་བང་ནི་ཐང་ལ་ཐས། འབངས་ཀྱིས་ནི་རྗེ་མ་གཏང། འབངས་ཀྱིས་ནི་རྗེ་བཏང་ན༎ འཇོན་མོ་ནི་ཤིང་ཀ་རེབས༎ སང་ཕྱི་ནི་གནངས་སླད་ན་ནམ་ནམ་ནི་ཞ་ཞ་རུ༎ རི་གང་ནི་ཁུས་བཏབ་ཀྱང། བྱ་པུ་ནི་ཁུས་མ་གདབ། ལུང་གང་ནི་རྟས་བཅགས་ཀྱང། གསེར་གད་ནི་རྟས་མ་བཅག ༎ ནི་འཆམ་འཆམ་ན༎ བཀའ་མཆིད་ནི་མིས་མི་མཉན༎ འཆམ་འཆམ་ནི་འདྲུལ་འདྲུལ་ན། ཆགས་ལམ་ནི་ཆུས་མི་སྦང། ཀླུ་དེ་ཞེས་བླངས་སོ༎ བླ་ན་རྗེ་སྒམ་ན། ཁྲི་སྲོང་བརྩན། འོག་ན་བློན་འཛངས་ན་སྟོང་རྩན་ཡུལ་ཟུང། རྗེ་ནི་གནམ་རི་པྱྭ་འི་ལུགས༎ བློན་པོ་ནི་ས་འི་ངམ་ལེན་གྱི་ཚུལ༎ མངའ་ཐང་ཆེན་པོ་འི་རྐྱེན་དུ། ཇི་དང་ཇིར་ལྡན་ཏེ། ཕྱི་འི་ཆབ་སྲིད་ནི་པྱོགས་བཞིར་བསྐྱེད༎ ནང་གི་ཁ་བསོ་ནི་མི་ཉམས་པར་ལྷུན་སྟུག ། འབངས་མགོ་ནག་པོ་ཡང་མཐོ་དམན་ནི་བསྙམས། དཔྱའ་སྒྱུ་ནི་བསྐྱུངས། དལ་དུ་ནི་མཆིས། སྟོན་དཔྱིད་ནི་བསྐྱལ༎ འཁོར་བར་ནི་སྤྱད། འདོད་པ་ནི་བྱིན། གནོད་པ་ནི་པྱེ། བཙན་བ་ནི་བཅུགས། སྡོ་བ་ནི་སྨད། འཇིགས་པ་ནི་མནན༎ བདེན་བ་ནི་བསྙེན། འཛངས་པ་ནི་བསྟོད། དཔའ་བོ་ནི་བཀུར། སྨོན་པར་ནི་བཀོལ༎ ཆོས་བཟང་སྲིད་མཐོ་སྟེ༎ མི་ཡོངས་ཀྱིས་སྐྱིད་དོ༎ བོད་ལ་སྔ་ན་ཡི་གེ་མེད་པ་ཡང༎ བཙན་པོ་འདི་འི་ཚེ་བྱུང་ནས༎ བོད་ཀྱི་གཙུགས་ལག་བཀའ་གྲིམས་ཆེད་པོ་དང། བློན་པོ་འི་རིམ་པ་དང། ཆེ་ཆུང་གཉིས་ཀྱི་དབང་ཐང་དང། ལེགས་པ་ཟིན་པ་འི་བྱ་དགའ་དང། ཉེ་ཡོ་བ་འི་ཆད་པ་དང། ཞིང་འབྲོག་གི་ཐུལ་ཀ་དང་དོར་ཀ་དང། སླུངས་ཀྱི་གོ་བར་བསྙམས་པ་དང། བྲེ་པུལ་དང། སྲང་ལ་སོགས་པ༎ བོད་ཀྱི་ཆོས་ཀྱི་གཞུང་བཟང་པོ་ཀུན༎ བཙན་པོ་ཁྲི་སྲོང་བརྩན་གྱི་རིང་ལས་བྱུང་ངོ། མི་ཡོངས་ཀྱིས་བཀའ་དྲིན་དྲན་ཞིང་ཚོར་བས༎
