སྐྱེ་ཟང་ལྡོང་ཚབ་བཀྱོན་ཕབ་ནས༎ དབས་སྣང་བཞེར་ཟླ་བརྩན་གྱིས་༎དེ་འི་འོག་དུ་མགོས་ཁྲི་བཟང་ཡབ་ལག་གིས་བྱས་སོ༎ དེ་འི་འོག་དུ། མཆིམས་ཞང་རྒྱལ་ཟིགས་ཤུ་ཏེང་གིས་བྱས་སོ༎ དེ་ནས་ངན་ལམ་སྟག་སྒྲ་ཀླུ་གོང་གིས་༎དེ་འི་འོག་དུ་སྣ་ནམ་ཞང་རྒྱལ་ཚན་ལྷ་སྣང་གིས་བྱས་སོ༎ དེ་འི་འོག་དུ་འབྲོ་ཁྲི་གཟུ་རམ་ཤགས་ཀྱིས་༎དེ་འི་འོག་དུ་དབའ་མང་རྗེ་ལྷ་ལོད་ཀྱིས་བྱས་སོ༎ དེ་འི་འོག་དུ་འབྲོ་ཁྲི་སུམ་རྗེ་སྟག་སྣང་གིས་བྱས་སོ༎ དེ་འི་འོག་དུ་དབས་རྒྱལ་ཏོ་རེ་སྟག་སྙས་བྱས་སོའ༎༆། ༔ ། མཁར་པྱིང་བ་སྟག་རྩེ་ན་ནི་རྒྱལ་སྟག་བུ་སྙ་གཟིགས་བཞུགས༎ ཉེན་ཀར་རྙིང་པ་ནི་ཟིང་པོ་རྗེ་སྟག་སྐྱ་བོ་མཆིས༎སྡུར་བ་འི་ཡུ་སྣ་ན་ནི༎ ཟིང་པོ་རྗེ་ཁྲི་པངས་སུམ་མཆིས༎འུང་ནས་ཟིང་པོ་རྗེ་སྟག་སྐྱ་བོ་ལྟ་ཞིག ༎ཇི་ལ་ཡང་ལོག་པ་གཤིན་པྱོགས་ཤེ་དག་བྱེད་དེ༎ ཉེས་པ་ནི་ལེགས་པར་བཅུག ། ལེགས་པ་ནི་ཉེས་པར་བཅུག་གོ༎འཛངས་པ་སྙིང་ཉེ་བ༎ལེགས་པ་འི་སྲིད་བྱས་བློ་བསྟན་པ་ནི་མི་ཉན༎མི་གཅམ་བུ་ངོ་ཟ་རི་བྱེད་པ་འི་ཚིག་ནི༎ ཆེད་དུ་ཡང་ཉན་ཏོ༎ མི་འདི་ལྟེ་བུ་འི་ཡུས་བྱུང་ངོ༎ མི་འཛངས་པ་དྲང་མཁྲང་བྱེད་པ་དང༎ དཔའ་བོ་ཆུ་གང་བྱེད་པ། ལ་སྡང་སྟེ། རི་མོ་བསྐྱུངས༎མ་འོས་པ་འི་ཆད་པ་ནི་ནན་ཆེ་འོ༎ མི་མཁོའ་བ་འི་ལས་ཀྱི་སྣ་གྲངས་ནི་མངས་སོ༎ འུང་ལྟར་ཟིང་པོ་རྗེ་སྟག་སྐྱ་བོ༎ ཆོས་ལོག་ཆིང་སྲིད་གྱུར་པས༎ རྗོ་བོ་ཉེས་དགུའ་བྱེད་པ༎མིག་གིས༎ མཐོང་ཡང༎ཉེས་པ་འི་བློ་གདབ་དུ་མ་འོས་པ་འི་ཆད་པ་ནན་ཆེས། ཚིག་དྲག་གིས༎ ཇི་ག་སུས་ཀྱང་བློ་གདབ། མ་ཕོད་དོ༎ རྗོ་བོ་ནི་བླ་ན་གཏུམ་མོ༎ ཁོལ་པོ་ནི་འོག་ན་སྐྲག་གོ༎རྗོ་བོ་ནི་བླ་ན་སྨྱོ། ཁོལ་པོ་ནི་འོག ༎ན་གྲམ་མོ༎ གཅིག་ལ་གཅིག་སྙིང་མི་བརྟན་ཏེ༎ ཐ་དད་དུ་སེམས་སོ༎ ཇོ་བོ་ཟིང་པོ་རྗེ་ངན་ཞན་ཏེ༎ ཇི་ལ་ཡང་ལོག་པ་གཤིན་པྱོགས་ཉེས་དགུར་བྱེད་པས༎ ཡུལ་གྱི་སྲིད་ཀྱང་དམས་སོ༎ མི་ཡོངས་ཀྱང་རྗོ་བོ། ལ་འཁང་ངོ༎འུང་ནས་ཟིང་པོ་རྗེ་སྟག་སྐྱ་བོ་འི་བློན་པོ༎མཉན་འཛི་ཟུང་ནག་པོས༎ རྗོ་བོ་ལ་གསོལ་པ༎ རྗོ་བོ་ཇི་ལ་ཡང་ལོག་པ་གཤིན་པྱོགས་མཛད་པས༎ སྐོལ་ཀྱི་སྲིད་ནི་སྲིད་དུ་མ་རུང།
