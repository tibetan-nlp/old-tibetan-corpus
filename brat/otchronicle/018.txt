འཛངས་པ་དང་དཔའ་བོ་འི་རི་མོ་བསྐྱེད་དོ༎ ངན་པ་མ་རབས་ནི་ཆིས་ཀྱིས་གསོས་སོའ༎ དེ་འི་ཚེ་བློན་པོ་སྲིད་བྱེད་པ་འི་རྣམས་ཀྱང། བློ་མཐུན་གྲོས་གཅིག་སྟེ༎ ཕྱི་འི་དགྲའ་བྱུང་བ་ལ། ཐབས་དང་ཡེ་མིག་ཆེར་བྱེད། ནང་གི་ཆོས་བྱ་བ་ལ་དྲང་ཞིང་འགྲུས་སུ་བྱེད༎ འཕྲག་མི་དོག ། ཉེས་མི་བྱེད༎ དཔའ་འཛངས་གཉིས་ནི་རླག་པ་བཞིན་བཙལ་ཏེ། འཛངས་པ་དཔའ་བོ་ནི་ཆེ་ས་ཆུང་སར་བསྟོད་ནས། ས་ས་ཡུལ་ཡུལ་དུ་བཀོད་དོ༎ འབངས་འོག་མ་དལ་ཞིང་ཡུལ་ན་འཁོད་པ་ནི༎ འཛངས་དྲང་གཉིས་སློབ་བའོ། དམག་མི་སོ་ལ་འཁོད་པ་ནི་དཔའ་བ་འི་ཐབས་དང། རྩལ་སློབ་བོ༎ བློ་སྒྱུ་གཉིས་ཀྱི་རླབས་ཆེད་པོས་ཐུབ་པས༎ སྡང་དགྲའ་སྡོ་བ་ཡང་མེད་དོ༎ སངས་རྒྱས་ཀྱི་ཆོས་བླ་ན་མེད་པ་བརྙེས་ཏེ་མཛད་ནས༎ དབུས་མཐའ་ཀུན་དུ་གཙུག་ལག་ཁང་བརྩིགས་ཏེ། ཆོས་བཙུགས་ནས། ཐམས་ཤད་ཀྱང་སྙིང་རྗེ་ལ་ཞུགས་ཤིང་དྲན་བས་སྐྱེ་ཤི་ལས་བསྒྲལ་ཏོ། གཡུང་དྲུང་དུ་བཏོན་ཏོ༎ རྗེ་བློན་ཏེ། ཞང་མཆིམས་རྒྱལ་ཟིགས་ལ་སོགས་པས་རྒྱ་འི་མཁར་ཀིང་ཤི་ཕབ་སྟེ། རྒྱ་རྗེ་གྭང་བུ་ཧྭང་ཏེ་བསྐོས་སོ༎ ལེགས་པ་ཞིན་པ་འི་བྱ་དགའ། གཡུ་འི་ཡི་གེ་ཆུ་ངུ་ནམ་ནམ་ཞར་ཞར་བྱིན་ནོ༎ དབས་སྐྱེས་བཟང་སྟག་སྣང་གིས༎ རྒྱ་འི་དམག་པོན་ཧོན་ཛེ་སངས་དང། འགུའ་ལོག་སྒང་དུ་གཡུལ་སྤྲད་ནས། རྒྱ་མང་པོ་བཏུངས་སྟེ། འགུའ་ལོག་རྒྱ་དུར་དུ་༎ སྦྲང་རྒྱལ་སྒྲ་ལེག་ཟིགས་ཀྱིས། སྟོད་ཕྱོགས་སུ་དྲངས་ཏེ། མུ་གཡུལ་བཟློག་ནས།  ལུང་གི་རྒྱལ་པོ་ནུང་ཀོག་མན་ཆད་འབངས་སུ་བསྡུས། དབས་བཙན་བཞེར་མདོ་ལོད་ལ་སོགས་པས། མཁར་ཚན་ཡན་ཆད་དུ་དྲངས་ཏེ། མཁར་ཅུ་པ་བརྒྱད་ཕབ་ནས། དོར་པོ་བཏོན་ཏེ། འབངས་སུ་བཞེས་སོ༎ ཆབ་སྲིད་ཆེ་སྟེ་ལོང་ཤན་ལ་རྒྱུད་ཡན་ཆད༎ ཕྱག་དུ་བཞེས་ནས། མཐོང་ཁྱབ་ཁྲི་སྡེ་ལྔ་བཙུགས། བདེ་བློན་ཁམས་ཆེད་པོ་གཅིག་གསར་དུ་བསྐྱེད་དོ༎ བློན་ཆེ་སྣང་བཞེར་ཟླ་བརྩན་འཛངས་ཤིང་མཁས་ལ། བཀྲའ་ཤིས་ཤིང་ཁ་དྲོ་བས། ནོར་བུ་རིན་པོ་ཆེ་འི་ཡི་གེ་སྩལ་ཏོ༎ འབངས་ཀྱི་ནང་ན། དོར་ཏེ་པྱུག་ཚམས་སྟེ་འཛོམ་དཔའ་བ་འི་མཚན་མར། ཐོག་བུ་སྩལ་ཏོ༎ བཙན་པོ་འི་ཞ་སྔ་ནས། ཐུགས་སྒམ་བ་འི་རླབས་དང། དབུ་རྨོག་བརྩན་པོ་འི་བྱིན་ཀྱིས༎ ཉི་ངོག་རྒྱལ་པོ་གཞན་དཔྱའ་ཕབ་པ་དང། རྒྱལ་པྲན་འབངས་སུ་བཀུག་པ་ལ་སོགས་པ་ཕྱིར༎
