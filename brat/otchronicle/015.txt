ངག་སྣ་ཆིག་འཚལ་རེའ༎ ཕུ་མ་ནུ་དང། བུ་ཚ་ལ་ལ་ཞིག་གིས༎ སྙིང་རིངས་སུ་བྱེད་ན༎ སྙིང་རིངས་སོ་ཞེས་ཀྱང་མི་འཐོལ་རེའ། པུ་ནུ་སྙིང་རིངས་པ་འི་གྲོགས་བྱེད་རེ༎ བཙན་པོ་ལ་གཞན་ཞིག་སྙིང་རིངས་པར་ཚོར་ན་ཡང། མི་འཐོལ་རེ། པུར་མི་མ་ནོངས་པ་ལ༎ འཕྲའ་མ་དང་འཕྲག་དོག་མཆི་རེ། དཔོན་དུ་བཅུག་ན། འབངས་སོ་ཆོག་ལ་སྙིང་སྙོམས་པར་མི་འཚལ་རེ༎ བཙན་པོས་བཀའ་ཇི་སྩལད་པ་གཟུངས་སུ་མི་འཛིན་རེ་ཞེས་བྲོ་སྩལད་ཏོ༎ བྲོ་བོར་བ་ནི། དབྱི་ཚབ་དངོས་དང། བུ་བསེ་དོ་རེ་ཉ་སྟོ། སྣང་ཏོ་རེ་སུམ་སྣང། ཁྲི་སུམ་རྗེ་ཅུང། མང་རྗེ་བཙན་ཟུང༎ ཚ་བོ་སྟག་པོ་རྗེ་མེས་སྣང། མང་པོ་རྗེ་པུ་ཚབ། དང་བདུན་གྱིས་བྲོ་བོར་རོའ༎ ༆༎ བཙན་པོ་སྲོང་བརྩན་སྒམ་པོའི་རིང་ལ༎ ཡབ་འབངས་ནི་འཁུས། ཡུམ་འབངས་ནི་ལོག ༎ གཉེན་ཞང་ཞུང། མཛོ་སུམ་པ༎ ཉག་ཉི་དག ས་པོ༎ རྐོང་པོ། མྱང་པོ་ཀུན་ཀྱང་ལོག ༎ ཡབ་གནམ་རི་སློན་མཚན་དུག་བོན་ཏེ་བཀྲོངས་སོ༎ སྲས་སྲོང་བརྩན་སྐུ་གཞོན་མ་ཕན་ཏེ༎ གཟོད་མ་དཀུའ་བ་དང། དུག་པ་རྣམས་རབས་བཅད་དོ༎ དེ་འི་རྗེས་ལ། དེ་འི་མི་ལོག་ཀུན་སླར་བཀུག་གོའ༎ འུང་གི་འོག་དུ་མྱང་མང་པོ་རྗེ་ཞང་སྣང་གིས། སུམ་པ་མཐའ་དག་དམགིས་གདབ་མི་དགོས་པར། ལུག་རྟུག་གིས་སྒྱུ་ཕབ་སྟེ། ལྕེ་འི་ཐོར་ཐོ་ལ་བརྗོད་ནས། ཁྱིམ་གྲངས་མ་ཤོར་པར་ཡོངས་ཀྱིས་འབངས་རྣལ་མར་བཀུག་གོ༎ འུང་གི་འོག་དུ་བཙན་པོ་ཞབས་ཀྱིས་བཙུགས་སྟེ༎ བྱང་ལམ་དུ་མ་བྱུང་མ་དྲངས་པར༎ རྒྱ་དང་འ་ཞས་དཔྱའ་གཅལ་ལོ༎ ཐོག་མ་འ་ཞ་དེ་ནས་འབངས་སུ་མངའ་འོ༎ འུང་གི་འོག་དུ་བཙན་པོ་སྲོང་བརྩན་དང། མང་པོ་རྗེ་ཞང་སྣང་རྗེས་འབངས་ཀྱི་བར་དུ༎ ཁྱུང་པོ་སྤུང་སད་ཀྱིས་འཕྲའ་མ་བཅུག་སྟེ། བཙན་པོ་ཞ་སྔར། ཞང་སྣང་གློ་བ་མི་འདྲིང་དུ་ནི་འདྲིང་ཞེས་གསོལ་ཏོ༎ ཞང་སྣང་ལ་བཙན་པོ་མི་བཀྱོན་དུ་བཀྱོན་ཞེས་བྱས་སོ༎ འུང་ནས། ཞང་སྣང་གི་གློ་བ་ལ། ཁྱུང་པོ་སྤུང་སད་བདག་དང་ཤག་རག་པོ་སྟེ། སྤུང་སད་ཟེར་བ་བདེན་སྙམ་ནས། བཙན་པོ་འི་བཀའ་བྱུང་ན་ཡང། ན་ཞེས་བྱ་སྟེ། མཁར་སྡུར་བར་ཐེན་ཏེ་མཆིར་མ་རུང་ངོ༎ འུང་ནས༎ བཙན་པོ་ཞ་སྔ་ནས༎ ཞང་སྣང་གློ་བ་འདྲིང་ཡང་དག་ཆེས་བཀའ་སྩལ་ནས། ཞང་སྣང་གི་མཁར་སྡུར་བ་ཡང་བསྡིགས་སོ༎
