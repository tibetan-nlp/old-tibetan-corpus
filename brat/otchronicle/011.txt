ཟུ་ཙེ་གློ་བ་ཉེ་བ་འི་བྱ་དགར་སྩལ་ཏོ༎ འུང་གི་འོག་དུ༎ བཙན་པོ་མཆེད་གཉིས་ལ༎ མོང་སྔོན་པོ་གློ་བ་རིངས་པ། ཟུ་ཙེ་གློ་བ་ཉེ་བས་དཀུའ་བེལ་ནས༎ བཙན་པོ་མཆེད་གཉིས་ཀྱི་སྐུ་ལ་མ་དར་པར༎ མོང་སྔོན་པོ་བཀུཾ་སྟེ༎ ཟུ་ཙེ་གློ་བ་ཉེ་འོ༎ ཟིང་པོ་རྗེ་སྲིད་བརླག་པ་འི་བློ་ལ། གཏོགས་ཏེ༎ ཟུ་ཙེ་གློ་བ་ཉེ་འོ༎ འུང་གི་རྗེས་ལ་དགས་པོ་འབངས་སུ་མངའ་བ་ལས་ལོག་གོ༎ འུང་ནས་རྗེ་བློན་ཡོངས་སུ་འདུས་ཏེ༎ དག ས་པོ་ལྷ་དེ་དགུག་པ་འི་བཀའ་གྲོས་མཛད་ནས། དམག་པོན་སུས་བྱ་ཞེས་བློད་ན༎ སེང་གོ་མི་ཆེན་གྱིས༎ ཁོ་བོས་རྔོ་ཐོག་ཆེས་ཁས་བླངས་སོ༎ འུང་ནས་ཁྱུང་པོ་སྤུང་སད་ན་རེ༎ ཁྱོ་འདའས། དམག་པོན་འོང་ངམ། མི་འཛངས་པ་གོ༎ སྒྱེ་འུ་ནང་དུ་སྨྱུང་བུ་བཅུག་པ་དང་མཚུངས་ཤེས་བྱ་ན༎ ཁྱོ་འདའ། བཙན་པོ་འི་སྣམ་ཕྱི་པར་བཀའ་སྩལ་ནས༎ ལོ་དུ་མ་ཞིག་ལོན་ན༎ འཛངས་རྔོ་ཐོག་གོ་ཞེས༎ མི་ཆིག་གིས། བསྟོད་པ་ཁོ་བོས་མ་ཐོས་ན༎ ཁྱོ་འདས་མི་འོང་བ་ལ་བསྒྲེ་བསྒྲེ་སྟེ༎ འབངས་ཆབ་འཚལ་བར་མཆི་འོ༎ ཞེས་བྱས་སོ༎ མི་ཆེན་ན་རེ༎ མི་ཡོངས་ཀྱིས་ཁོ་བོ་ལ་སྙན་དུ་མི་བརྗོད་པ་ཡང་བདེན་ནོ༎ ཁོ་བོ་སྔོན་སྒྱེ་འུ་ནང་དུ་ཞུགས་ཀྱང་མ་ཞུགས། རྩེ་མོ་པྱིར་མ་བྱུང་བ་ཡང་བདེན་ནོ༎ ཁོ་བོ་སྔོན་སྒྱེ་འུ་ནང་དུ་ཞུགས་སུ་ཟིན་ན༎ རྩེ་མོ་བས་ཡུ་བ་ཕན་ཆད་པྱུང་སྟེ་སོང་ཟིན་ནོ༎ འུང་གིས་དེང་ཁོ་བོས་གསོལ་བ་ཡང༎ སྔོན་ཇི་འི་ནང་དུ་ཡང་མ་ཞུགས་པས༎ གདོད་འཇུག་པར་གསོལ་བ་ཡིན་ནོ་ཞེས་བྱས་སོ༎ འུང་ནས༎ བཙན་པོས་ཀྱང། མི་ཆེན་གསོལ་བ་བཞིན་གནང་ནས༎ དག ས་པོ་དགུག་པ་འི་དམག་པོན་དུ་བཀའ། སྩལ་ཏོ༎ འུང་ནས་མི་ཆེན་གྱིས༎ དག ས་པོ་ལྷ་དེ་ལ་བརྒལ་ཏེ༎ དག ས་པོ་ཡོངས་སུ་བཀུག་སྟེ༎ མི་ཆེན་གྱི་བྱ་དགའར༎ གསེར་ཁུང་རེ་འུ་རྒལ་གྱི་མི་དང་འབྲོག་དུ་བཅས་ཏེ་སྩལ་ཏོ༎ འབྲོང་ཆེ་པོ་རྔུལ་གྱི་རྭ་མི། ཡོངས་ཀྱི་ཁ་ན་བརྗོད་ན༎ སོང་འགོ་མི་ཆེན་གྱིས༎ དགས་ཀྱི་བསེན་མཁར་ལ༎ མདུང་མོ་ཐབས། གཅིག་གིས། འབྲོག་མོ་ཐང་གང་ཐོབ༎ པྱོགས་རྔེག་སུམ་བརྩན་སྐྱ་བོ་རེང་གིས། དཔའ་བ་འི་པུལ་པྱིན་ན༎ ཉིན་ཆིག་ལ་དགས་པོ་མི་བརྒྱའ་བསད་ཆེས་འུང་སྐད་དོ༎ འུང་གི་རྗེས་ལ། མྱང་སྨོན་ཏོ་རེ་གློ་བ་ཉེ་བའི་བུ༎ ཞང་སྣང་བཙན་པོ་འི་ཞ་འབྲིང་སྣང་མ་པ་འཚལ་ཞིང་མཆིས་སོ༎ འུང་གི་རྗེས་ལ༎ བཙན་པོ་རྗེས་འབངས་དགྱེས་སྐྱེམས་སྟོན་མོ་གསོལ་ལོ༎ འུང་གི་ཚེ་ཁྱུང་པོ་སྤུང་སད་ཀྱིས་ཀླུ་བླངས་པ་འི༎ ཚིག་ལ༎ མོན་ཀ་འི་ནི་སྟག་ཆིག་པ། སྟག་བཀུམ་ནི་ཟུ་ཙེས་བཀུམ། གུང་བཀྲོས་ནི་ཕྱག་དུ་པུལ། སླ་ལྦོ་ནི། ལྷེ་རྔེགས་སྩལད༎ རྩང་བྲང་ནི་ཡ་སྟོད་ཀྱི་ཐང་པྲོམ་ནི་རྒོད་ལྡིང་བ༎ རྒོད་བཀུཾ་ནི་ཟུ་ཙེས་བཀུམ༎ རྒོད་གཤོག་ནི་ཕྱག་དུ་པུལ། གསབ་གསོབ་ནི་ལྷོ་རྔེགས་སྩལད༎ ན་ནིང་ནི་གཞེ་ནིང་སྔ༎ ཏི་སེ་ནི་གངས་དྲུང་ནས༎ ཤ་དང་ནི་རྐྱང་བྱེར་བ༎ ཤམ་པོ་ནི་རྩ་ལ་བྱེར༎ དི་རིང་ནི་སང་ལྟ་ན་ཤམ་པོ་ནི་གཉན་གྱི་རྩ༎ ཤ་རྐྱང་ནི་ཚས་མ་ངན༎ ཤ་རྐྱང་ནི་ཚས་ངན་ན༎ ཏི་སེ་ནི་གངས་ཀྱིས་བྲུན༎ ན་ནིང་ནི་གཞེ་ནིང་སྔོན། མ་པང་ནི་མཚོ་འགྲམ་ནས། ངང་དང་ནི་ངུར་བྱེར་བ༎ དང་ཀོ་ནི་མཚོ་ལ་བྱེར༎ དི་རིང་ནི་སང་ལྟ་ན། དང་ཀོ་ནི་ལྷ་འི་མཚོའ༎ ང་ངུར་ནི་ཚས་མ་ངན། ང་ངུར་ནི་ཚས་ངན་ན། མ་པང་ནི་མཚོ་འིས་བྲུན༎ ལྷོ་རྔེགས་ནི་འཕན་གྱི་སྣོན། སེ་ཁྱུང་ནི་འཕན་གྱིས་བཏབ༎ སྔ་ན་ནི་འཕན་བ་ལ། ད་ཙམ་ནི་སྤྱན་ཡང་ཡས། དབུ་པྱིང་ནི་གྲོ་བོ་ལ། ཐ་མ་ནི་གཡགིས་བསྐོརད། ཟུ་ཙེ་ནི་སྩལ་ལགས་ཀྲང། ཀླུ་དེ་ཞེས་༎ འུང་ནས་བཙན་པོ་འི་ཐུགས་ལ་དགོངས་པ༎ བློན་པོ་ལྷོ་རྔེགས་གཅིག་གིས་ལན་བླན་བར་ཐུགས་རེ་བ་ལས། སུས་ཀྱང་མ་བླན་ཏོ༎ འུང་ནས་ཞང་སྣང་པེ་འུ་ཟུར་འབྲིང་པོ་བཅས་པ་ཙམ་ཞིག ༎ སྣང་མ་ལྟོམ་ཞིང་མཆིས་པ་ལ༎ ཁྱོད་ཕ་སྙིང་ཉེ་བ་འི་བུ་ན༎ ཀླུ་ཤེས་སམ་ཞེས་རྨས་པ་དང༎ ཞང་སྣང་གིས་མཆིད་འཚལ་ཞེས་གསོལ་ཏོ༎ ཞང་སྣང་གིས་ཀླུ་བླངས་པ༎ ཀྱེ་ན་ནིང་ནི་གཞེ་ནིང་སྔ༎ ཆབ་ཆབ་ནི་ཕ་རོལ་ན༎ ཡར་ཆབ་ནི་ཕ་རོལ་ན། དགུ་གྲི་ནི་ཟིང་པོ་རྗེ༎ ཉ་ལས་ནི་ཁམ་དུ་བཀོད། ཁམ་ལས་ནི་རྨེག་དུ་བཏའ༎ སྐྱི་ནས་ནི་ཉ་འདོན་པ། དབྱི་ཚབ་ནི་པངས་ཏོ་རེ༎ ཀླུམ་ན་ནི་ཆབ་གཅོད་པ། ཙེང་སྐུ་ནི་སྨོན་ཏོ་རེ། ཁུར་ར་ནི་རགས་ཐོགས་ཤིང་དོག་གི་ནི་ས་སྐྱེད་པ༎
